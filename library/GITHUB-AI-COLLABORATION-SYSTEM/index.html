    .github/workflows/ai-task-handler.yml¶

    name: AI Task Handler

on:
  issues:
    types: [opened, labeled]
  issue_comment:
    types: [created]

permissions:
  contents: write
  issues: write
  pull-requests: write

jobs:
  process-ai-task:
    if: |
      (github.event.action == 'opened' && contains(github.event.issue.labels.*.name, 'ai-task')) ||
      (github.event.action == 'labeled' && github.event.label.name == 'ai-task')
    runs-on: ubuntu-latest
    steps:
      - name: Acknowledge task
        uses: actions/github-script@v7
        with:
          script: |
            await github.rest.issues.createComment({
              owner: context.repo.owner,
              repo: context.repo.repo,
              issue_number: context.issue.number,
              body: `ü§ñ **AI Task Acknowledged**\n\nI'm analyzing the requirements and will start work shortly.\n\n**Status:** üîÑ Processing\n**Started:** ${new Date().toISOString()}\n**Estimated completion:** TBD (will update in 15 minutes)\n\nYou'll receive mobile notifications for:\n- ‚úÖ Task completion\n- ü§î Decisions needed\n- üö® Blockers encountered\n\nTracking this in Command Center: #4`
            });

            await github.rest.issues.addLabels({
              owner: context.repo.owner,
              repo: context.repo.repo,
              issue_number: context.issue.number,
              labels: ['in-progress', 'ai-working']
            });

      - name: Notify Command Center
        uses: actions/github-script@v7
        with:
          script: |
            await github.rest.issues.createComment({
              owner: context.repo.owner,
              repo: context.repo.repo,
              issue_number: 4, // Command Center issue
              body: `üÜï **New AI Task Started**\n\n**Task:** #${context.issue.number} ${context.payload.issue.title}\n**Priority:** ${context.payload.issue.labels.find(l => l.name.startsWith('priority'))?.name || 'Not set'}\n**Status:** üîÑ Processing\n\n[View Task ‚Üí](${context.payload.issue.html_url})`
            });

  process-approval:
    if: |
      github.event.action == 'created' &&
      contains(github.event.issue.labels.*.name, 'needs-approval') &&
      (contains(github.event.comment.body, 'üëç') || contains(github.event.comment.body, 'üëé'))
    runs-on: ubuntu-latest
    steps:
      - name: Handle approval
        uses: actions/github-script@v7
        with:
          script: |
            const approved = context.payload.comment.body.includes('üëç');
            const action = approved ? 'approved' : 'rejected';
            const emoji = approved ? '‚úÖ' : '‚ùå';

            await github.rest.issues.createComment({
              owner: context.repo.owner,
              repo: context.repo.repo,
              issue_number: context.issue.number,
              body: `${emoji} **Decision ${action.toUpperCase()} by @${context.payload.comment.user.login}**\n\nI'm proceeding with the ${approved ? 'approved' : 'alternative'} approach.\n\n**Next steps:**\n${approved ? '- Implementing approved solution\n- Will report progress in 2 hours\n- Estimated completion: [TIME]' : '- Analyzing alternative options\n- Will present new proposal in 1 hour'}\n\nTracking in Command Center: #4`
            });

            if (approved) {
              await github.rest.issues.addLabels({
                owner: context.repo.owner,
                repo: context.repo.repo,
                issue_number: context.issue.number,
                labels: ['approved', 'ai-working']
              });
              await github.rest.issues.removeLabel({
                owner: context.repo.owner,
                repo: context.repo.repo,
                issue_number: context.issue.number,
                name: 'needs-approval'
              });
            }

    .github/workflows/command-center-daily-brief.yml¶

    name: Command Center Daily Brief

on:
  schedule:
    - cron: "0 13 * * *" # 8:00 AM EST (13:00 UTC)
  workflow_dispatch:

permissions:
  contents: read
  issues: write

jobs:
  post-daily-brief:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Generate briefing
        id: brief
        run: |
          # Read latest metrics
          METRICS=$(cat metrics/latest-metrics.json 2>/dev/null || echo '{}')

          # Count issues by status
          COMPLETED=$(gh issue list --label "done" --search "created:>=$(date -d '1 day ago' +%Y-%m-%d)" --json number --jq '. | length')
          IN_PROGRESS=$(gh issue list --label "in-progress" --json number --jq '. | length')
          BLOCKED=$(gh issue list --label "blocked" --json number --jq '. | length')

          echo "completed=$COMPLETED" >> $GITHUB_OUTPUT
          echo "in_progress=$IN_PROGRESS" >> $GITHUB_OUTPUT
          echo "blocked=$BLOCKED" >> $GITHUB_OUTPUT
        env:
          GH_TOKEN: ${{ secrets.GITHUB_TOKEN }}

      - name: Post briefing to Command Center
        uses: actions/github-script@v7
        with:
          script: |
            const date = new Date().toISOString().split('T')[0];
            await github.rest.issues.createComment({
              owner: context.repo.owner,
              repo: context.repo.repo,

    ü§ñ AI Agent Delegation Framework¶

    How AI Spawns Sub-Agents¶

    Pattern 1: P02 Requirements Agent (Already Active) 

    1. Marco requests: "Build voice demo"
2. AI creates P02 task JSON with comprehensive requirements
3. P02 agent processes autonomously (2-4 hours)
4. P02 posts deliverables to GitHub
5. AI reviews, submits to Marco for approval via issue comment
6. Marco approves (üëç), AI proceeds to implementation

    Pattern 2: Dev Team Agent (For Implementation) 

    1. AI receives approved requirements from P02
2. AI creates GitHub issue: "[AI-TASK] Implement useVoiceInteraction hook"
3. AI assigns to sub-agent (simulated via runSubagent tool)
4. Sub-agent:
   - Reads requirements
   - Writes code
   - Creates tests
   - Opens PR
   - Reports back to parent AI
5. Parent AI reviews PR, merges if tests pass
6. Posts completion report to Command Center (#4)

    Pattern 3: QA Agent (For Testing) 

    1. After implementation complete, AI spawns QA agent
2. QA agent:
   - Runs automated tests (unit, integration, E2E)
   - Performs accessibility audit (Lighthouse, axe)
   - Tests on multiple browsers
   - Validates WCAG compliance
   - Generates test report
3. If tests fail:
   - QA agent creates bug issues
   - Dev agent fixes bugs
   - QA agent re-tests
4. If tests pass:
   - QA agent generates Marco's UAT script
   - Posts to Command Center with deployment URL

    Pattern 4: DevOps Agent (For Deployment) 

    1. After QA approval, AI spawns DevOps agent
2. DevOps agent:
   - Builds production bundle
   - Runs security scan
   - Deploys to GitHub Pages
   - Runs smoke tests on production
   - Generates deployment report
3. Posts live URL to Command Center
4. Monitors for errors (first 30 minutes)
