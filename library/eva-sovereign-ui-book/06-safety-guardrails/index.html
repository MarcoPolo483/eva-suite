    Chapter 06: Safety & Guardrails in EVA Chat 2.0¶

    Overview¶

    Safety and guardrails are non-negotiable requirements for AI systems deployed in Government of Canada environments. EVA Chat 2.0 implements a defense-in-depth strategy with multiple layers of protection:

    1. Content Safety: Azure Content Safety API for PII detection, hate speech, violence, self-harm
    2. C0-C3 Autonomy Levels: GC Agile Agentic Framework compliance with explicit human oversight gates
    3. Output Validation: Hallucination detection, fact-checking against RAG sources, confidence thresholding
    4. Protected B Compliance: Encryption at rest/in-transit, audit logging, data residency enforcement
    5. UI Disclaimers: eva-gc-ai-disclaimer component for transparency and user consent

    This chapter covers the architecture, implementation, and compliance validation for each safety layer.

    Content Safety with Azure Content Safety API¶

    Azure Content Safety Integration¶

    EVA Chat 2.0 uses Azure AI Content Safety to detect harmful content in both user inputs and AI-generated responses:

    import { ContentSafetyClient } from '@azure/ai-content-safety';
import { DefaultAzureCredential } from '@azure/identity';

interface SafetyAnalysisResult {
  passed: boolean;
  severity: 'safe' | 'low' | 'medium' | 'high';
  categories: {
    hate: number;
    sexual: number;
    violence: number;
    selfHarm: number;
  };
  piiDetected: boolean;
  blockedReason?: string;
}

class ContentSafetyService {
  private client: ContentSafetyClient;
  private readonly SEVERITY_THRESHOLD = 4;  // Block medium/high (4-7), allow low (0-3)

  constructor() {
    const credential = new DefaultAzureCredential();
    this.client = new ContentSafetyClient(
      process.env.AZURE_CONTENT_SAFETY_ENDPOINT!,
      credential
    );
  }

  async analyzeText(text: string): Promise<SafetyAnalysisResult> {
    const response = await this.client.analyzeText({
      text,
      categories: ['Hate', 'Sexual', 'Violence', 'SelfHarm'],
      outputType: 'EightSeverityLevels',  // 0-7 scale
      haltOnBlocklistHit: true
    });

    const maxSeverity = Math.max(
      response.categoriesAnalysis.find(c => c.category === 'Hate')?.severity || 0,
      response.categoriesAnalysis.find(c => c.category === 'Sexual')?.severity || 0,
      response.categoriesAnalysis.find(c => c.category === 'Violence')?.severity || 0,
      response.categoriesAnalysis.find(c => c.category === 'SelfHarm')?.severity || 0
    );

    // Detect PII (separate API call)
    const piiResult = await this.detectPII(text);

    const passed = maxSeverity < this.SEVERITY_THRESHOLD && !piiResult.detected;

    return {
      passed,
      severity: this.mapSeverity(maxSeverity),
      categories: {
        hate: response.categoriesAnalysis.find(c => c.category === 'Hate')?.severity || 0,
        sexual: response.categoriesAnalysis.find(c => c.category === 'Sexual')?.severity || 0,
        violence: response.categoriesAnalysis.find(c => c.category === 'Violence')?.severity || 0,
        selfHarm: response.categoriesAnalysis.find(c => c.category === 'SelfHarm')?.severity || 0
      },
      piiDetected: piiResult.detected,
      blockedReason: passed ? undefined : this.getBlockedReason(maxSeverity, piiResult)
    };
  }

  private async detectPII(text: string): Promise<{ detected: boolean; entities: string[] }> {
    const piiResponse = await this.client.analyzePII({
      text,
      domain: 'PHI',  // Protected Health Information
      categories: [
        'Address', 'Email', 'Name', 'PhoneNumber', 'SSN', 'CreditCard', 'IPAddress'
      ]
    });

    const detected = piiResponse.categoriesAnalysis.length > 0;
    const entities = piiResponse.categoriesAnalysis.map(c => c.category);

    return { detected, entities };
  }

  private mapSeverity(level: number): 'safe' | 'low' | 'medium' | 'high' {
    if (level === 0) return 'safe';
    if (level <= 3) return 'low';
    if (level <= 5) return 'medium';
    return 'high';
  }

  private getBlockedReason(severity: number, piiResult: { detected: boolean; entities: string[] }): string {
    if (piiResult.detected) {
      return `Personal information detected: ${piiResult.entities.join(', ')}`;
    }
    if (severity >= 6) {
      return 'High-severity harmful content detected';
    }
    return 'Medium-severity harmful content detected';
  }
}

    Input Validation Workflow¶

    User Input Content Safety Check Allow/Block

    class ChatInputValidator {
  private safetyService: ContentSafetyService;
  private auditLogger: AuditLogger;

  async validateUserInput(userId: string, message: string): Promise<ValidationResult> {
    // Run content safety analysis
    const safetyResult = await this.safetyService.analyzeText(message);

    // Log validation attempt (even if blocked)
    await this.auditLogger.logValidation({
      userId,
      messageLength: message.length,
      passed: safetyResult.passed,
      severity: safetyResult.severity,
      piiDetected: safetyResult.piiDetected,
      timestamp: new Date()
    });

    if (!safetyResult.passed) {
      // Block message and notify user
      return {
        allowed: false,
        reason: safetyResult.blockedReason,
        displayMessage: this.getSafeErrorMessage(safetyResult)
      };
    }

    return { allowed: true };
  }

  private getSafeErrorMessage(result: SafetyAnalysisResult): string {
    if (result.piiDetected) {
      return 'Your message contains personal information. Please remove sensitive details and try again.';
    }

    return 'Your message violates our content policy. Please rephrase and try again.';
  }
}

    Output Validation Workflow¶

    AI Response Content Safety Check Redact/Allow

    class ChatOutputValidator {
  private safetyService: ContentSafetyService;

  async validateAssistantResponse(response: string): Promise<string> {
    const safetyResult = await this.safetyService.analyzeText(response);

    if (!safetyResult.passed) {
      // Log incident (AI-generated harmful content is a critical issue)
      await this.auditLogger.logCriticalIncident({
        type: 'AI_HARMFUL_CONTENT',
        content: response,
        severity: safetyResult.severity,
        timestamp: new Date()
      });

      // Return safe fallback response
      return 'I apologize, but I cannot provide a response to that question. Please contact support for assistance.';
    }

    // Redact PII if detected
    if (safetyResult.piiDetected) {
      return await this.redactPII(response);
    }

    return response;
  }

  private async redactPII(text: string): Promise<string> {
    // Use Azure Content Safety PII redaction
    const redactedResponse = await this.safetyService.client.redactPII({
      text,
      domain: 'PHI',
      redactionFormat: 'PlainText'
    });

    return redactedResponse.redactedText;
  }
}

    C0-C3 Autonomy Levels: GC Agentic Framework¶

    Autonomy Level Definitions¶

    The Government of Canada Agentic AI Framework defines four autonomy levels for AI systems:

    Level Name Description Human Oversight EVA Chat 2.0 Usage
    C0 Human-in-the-Loop AI provides suggestions; human makes all decisions Continuous Default for all responses
    C1 Human-on-the-Loop AI acts with human approval for critical actions Approval gates High-confidence RAG responses
    C2 Human-over-the-Loop AI acts autonomously; human can override Monitoring Routine FAQ responses
    C3 Fully Autonomous AI acts without human intervention Audit only NOT USED in EVA Chat 2.0

    EVA Chat 2.0 Policy: ALL responses are tagged with autonomy level. C3 is never used (prohibited for Protected B systems).

    Autonomy Level Detection¶

    interface AutonomyMetadata {
  level: 'C0' | 'C1' | 'C2';
  confidence: number;
  requiresApproval: boolean;
  rationale: string;
}

class AutonomyLevelService {
  determineLevel(response: string, citations: Citation[]): AutonomyMetadata {
    const avgConfidence = this.calculateAvgConfidence(citations);

    // C2: High-confidence, well-cited, routine questions
    if (avgConfidence >= 0.8 && citations.length >= 3 && this.isRoutineQuestion(response)) {
      return {
        level: 'C2',
        confidence: avgConfidence,
        requiresApproval: false,
        rationale: 'High-confidence response with multiple authoritative sources'
      };
    }

    // C1: Medium-confidence, requires human validation
    if (avgConfidence >= 0.6 && citations.length >= 2) {
      return {
        level: 'C1',
        confidence: avgConfidence,
        requiresApproval: true,
        rationale: 'Medium-confidence response; human validation recommended'
      };
    }

    // C0: Low-confidence or no citations - human must verify
    return {
      level: 'C0',
      confidence: avgConfidence,
      requiresApproval: true,
      rationale: 'Low-confidence response; treat as suggestion only'
    };
  }

  private isRoutineQuestion(response: string): boolean {
    // Check if response matches pre-approved FAQ patterns
    const faqPatterns = [
      /how to apply for (EI|CPP|OAS)/i,
      /what is the eligibility/i,
      /contact information for/i
    ];

    return faqPatterns.some(pattern => pattern.test(response));
  }

  private calculateAvgConfidence(citations: Citation[]): number {
    if (citations.length === 0) return 0;
    const total = citations.reduce((sum, c) => sum + c.confidence, 0);
    return total / citations.length;
  }
}

    Autonomy Level Enforcement¶

    class ResponseGuardrailService {
  async enforceAutonomyLevel(
    response: string,
    autonomy: AutonomyMetadata,
    userId: string
  ): Promise<GuardrailResult> {
    // C0: Always show disclaimer, require user acknowledgment
    if (autonomy.level === 'C0') {
      return {
        allowed: true,
        disclaimer: 'This response is AI-generated and may not be accurate. Verify with official sources.',
        requiresAcknowledgment: true
      };
    }

    // C1: Require human approval for actionable responses
    if (autonomy.level === 'C1' && this.isActionable(response)) {
      return {
        allowed: false,  // Hold for approval
        disclaimer: 'This response contains actionable advice. Please review before acting.',
        requiresApproval: true,
        approvalWorkflow: 'senior-analyst-review'
      };
    }

    // C2: Allow with standard disclaimer
    return {
      allowed: true,
      disclaimer: 'This response is based on official Government of Canada sources.',
      requiresAcknowledgment: false
    };
  }

  private isActionable(response: string): boolean {
    const actionKeywords = [
      'submit', 'apply', 'file', 'contact', 'send', 'call', 'visit', 'complete'
    ];

    return actionKeywords.some(keyword =>
      response.toLowerCase().includes(keyword)
    );
  }
}

    Output Validation: Hallucination Detection¶

    Hallucination Detection Strategy¶

    Problem: LLMs can generate plausible-sounding but factually incorrect information.

    Solution: Cross-reference AI responses with RAG citations to detect unsupported claims.

    interface HallucinationCheckResult {
  isGrounded: boolean;
  unsupportedClaims: string[];
  confidence: number;
}

class HallucinationDetector {
  private openai: OpenAIClient;

  async detectHallucinations(
    response: string,
    citations: Citation[]
  ): Promise<HallucinationCheckResult> {
    // Extract claims from response
    const claims = await this.extractClaims(response);

    // Verify each claim against citations
    const verificationResults = await Promise.all(
      claims.map(claim => this.verifyClaim(claim, citations))
    );

    const unsupportedClaims = verificationResults
      .filter(r => !r.supported)
      .map(r => r.claim);

    const confidence = verificationResults.filter(r => r.supported).length / claims.length;

    return {
      isGrounded: unsupportedClaims.length === 0,
      unsupportedClaims,
      confidence
    };
  }

  private async extractClaims(response: string): Promise<string[]> {
    const prompt = `
Extract factual claims from the following text. Return a JSON array of strings.
Only include objective, verifiable statements (not opinions or questions).

Text: ${response}

Output format: ["claim1", "claim2", ...]
`;

    const result = await this.openai.getChatCompletions(
      process.env.AZURE_OPENAI_DEPLOYMENT!,
      [{ role: 'user', content: prompt }],
      { temperature: 0, responseFormat: { type: 'json_object' } }
    );

    const parsed = JSON.parse(result.choices[0].message.content);
    return parsed.claims || [];
  }

  private async verifyClaim(claim: string, citations: Citation[]): Promise<{ claim: string; supported: boolean }> {
    // Combine all citation excerpts
    const citationText = citations.map(c => c.excerpt).join('\n\n');

    const prompt = `
Does the following claim have support in the provided sources?
Answer with "SUPPORTED" or "UNSUPPORTED" only.

Claim: ${claim}

Sources:
${citationText}
`;

    const result = await this.openai.getChatCompletions(
      process.env.AZURE_OPENAI_DEPLOYMENT!,
      [{ role: 'user', content: prompt }],
      { temperature: 0, maxTokens: 10 }
    );

    const answer = result.choices[0].message.content.trim();
    return { claim, supported: answer === 'SUPPORTED' };
  }
}

    Fact-Checking Workflow¶

    class ResponseFactChecker {
  private hallucinationDetector: HallucinationDetector;

  async validateResponse(
    response: string,
    citations: Citation[]
  ): Promise<FactCheckResult> {
    // Detect hallucinations
    const hallucinationCheck = await this.hallucinationDetector.detectHallucinations(
      response,
      citations
    );

    // If unsupported claims found, flag response
    if (!hallucinationCheck.isGrounded) {
      return {
        passed: false,
        issues: hallucinationCheck.unsupportedClaims,
        action: 'regenerate',  // Force regeneration with stricter grounding
        disclaimer: 'This response contains unverified claims. Please review with caution.'
      };
    }

    // Check confidence threshold
    if (hallucinationCheck.confidence < 0.7) {
      return {
        passed: true,
        issues: [],
        action: 'warn',
        disclaimer: 'This response has moderate confidence. Verify with official sources.'
      };
    }

    return {
      passed: true,
      issues: [],
      action: 'allow',
      disclaimer: null
    };
  }
}

    Protected B Compliance¶

    Encryption at Rest and In Transit¶

    Azure Key Vault manages all encryption keys (Customer-Managed Keys - CMK):

    import { SecretClient } from '@azure/keyvault-secrets';
import { CryptographyClient } from '@azure/keyvault-keys';

class EncryptionService {
  private secretClient: SecretClient;
  private cryptoClient: CryptographyClient;

  constructor() {
    const credential = new DefaultAzureCredential();
    this.secretClient = new SecretClient(
      process.env.AZURE_KEYVAULT_URL!,
      credential
    );
    this.cryptoClient = new CryptographyClient(
      process.env.AZURE_KEYVAULT_KEY_URL!,
      credential
    );
  }

  async encryptConversation(conversation: Conversation): Promise<EncryptedConversation> {
    // Serialize conversation
    const plaintext = JSON.stringify(conversation);

    // Encrypt using CMK
    const { result } = await this.cryptoClient.encrypt({
      algorithm: 'RSA-OAEP-256',
      plaintext: Buffer.from(plaintext, 'utf-8')
    });

    return {
      id: conversation.id,
      userId: conversation.userId,
      encryptedData: result.toString('base64'),
      keyVersion: process.env.AZURE_KEYVAULT_KEY_VERSION!,
      timestamp: new Date()
    };
  }

  async decryptConversation(encrypted: EncryptedConversation): Promise<Conversation> {
    const { result } = await this.cryptoClient.decrypt({
      algorithm: 'RSA-OAEP-256',
      ciphertext: Buffer.from(encrypted.encryptedData, 'base64')
    });

    return JSON.parse(result.toString('utf-8'));
  }
}

    Audit Logging for Protected B¶

    Requirement: All access to Protected B data MUST be logged with 7-year retention.

    interface AuditEvent {
  eventType: 'READ' | 'WRITE' | 'DELETE' | 'SEARCH';
  userId: string;
  resourceId: string;
  resourceType: 'conversation' | 'user-profile' | 'document';
  classification: 'unclassified' | 'protected-a' | 'protected-b';
  ipAddress: string;
  userAgent: string;
  timestamp: Date;
  success: boolean;
  errorMessage?: string;
}

class ProtectedBAuditLogger {
  private cosmosClient: CosmosClient;
  private readonly RETENTION_YEARS = 7;

  async logAccess(event: AuditEvent): Promise<void> {
    const ttl = this.RETENTION_YEARS * 365 * 24 * 60 * 60;  // 7 years in seconds

    await this.cosmosClient
      .database('eva-audit')
      .container('access-logs')
      .items.create({
        ...event,
        ttl,  // Cosmos DB auto-deletes after 7 years
        partitionKey: event.userId
      });
  }

  async queryAccessLogs(
    userId: string,
    startDate: Date,
    endDate: Date
  ): Promise<AuditEvent[]> {
    const query = `
      SELECT * FROM c
      WHERE c.userId = @userId
      AND c.timestamp >= @startDate
      AND c.timestamp <= @endDate
      ORDER BY c.timestamp DESC
    `;

    const { resources } = await this.cosmosClient
      .database('eva-audit')
      .container('access-logs')
      .items.query({
        query,
        parameters: [
          { name: '@userId', value: userId },
          { name: '@startDate', value: startDate.toISOString() },
          { name: '@endDate', value: endDate.toISOString() }
        ]
      })
      .fetchAll();

    return resources;
  }
}

    Data Residency Enforcement¶

    Requirement: All Protected B data MUST reside in Canada Central or Canada East regions.

    class DataResidencyValidator {
  private allowedRegions = ['canadacentral', 'canadaeast'];

  async validateResourceLocation(resourceId: string): Promise<boolean> {
    // Check Cosmos DB location
    const cosmosAccount = await this.getCosmosAccount();
    const locations = cosmosAccount.readLocations.map(l => l.locationName.toLowerCase());

    const allInCanada = locations.every(loc => this.allowedRegions.includes(loc));

    if (!allInCanada) {
      await this.auditLogger.logCriticalIncident({
        type: 'DATA_RESIDENCY_VIOLATION',
        resourceId,
        locations,
        timestamp: new Date()
      });

      throw new Error('Data residency violation: resource outside Canada');
    }

    return true;
  }

  async enforceRegionPolicy(): Promise<void> {
    // Azure Policy assignment to block non-Canada deployments
    const policyDefinition = {
      displayName: 'Enforce Canada Data Residency',
      policyRule: {
        if: {
          not: {
            field: 'location',
            in: ['canadacentral', 'canadaeast']
          }
        },
        then: {
          effect: 'deny'
        }
      }
    };

    // Apply via Azure SDK
    await this.azurePolicyClient.createOrUpdate('canada-residency-policy', policyDefinition);
  }
}

      ${this.requiresAcknowledgment ? html`
        <div class="acknowledgment-checkbox">
          <input
            type="checkbox"
            id="acknowledge"
            @change=${this.handleAcknowledge}
            ?checked=${this.acknowledged}
          />
          <label for="acknowledge">${text.acknowledgmentText}</label>
        </div>
      ` : ''}
    `;
  }

  private getLocalizedText() {
    const isFrench = this.language === 'fr-CA';

    const autonomyLabels = {
      'C0': isFrench ? 'Humain dans la boucle' : 'Human-in-the-Loop',
      'C1': isFrench ? 'Humain sur la boucle' : 'Human-on-the-Loop',
      'C2': isFrench ? 'Humain au-dessus de la boucle' : 'Human-over-the-Loop'
    };

    const messages = {
      'C0': isFrench
        ? 'Cette rponse est gnre par l\'IA et peut ne pas tre exacte. Vrifiez avec des sources officielles.'
        : 'This response is AI-generated and may not be accurate. Verify with official sources.',
      'C1': isFrench
        ? 'Cette rponse contient des conseils pratiques. Veuillez rviser avant d\'agir.'
        : 'This response contains actionable advice. Please review before acting.',
      'C2': isFrench
        ? 'Cette rponse est base sur des sources officielles du gouvernement du Canada.'
        : 'This response is based on official Government of Canada sources.'
    };

    return {
      title: isFrench ? 'Avis de l\'IA' : 'AI Notice',
      message: messages[this.autonomyLevel],
      autonomyLabel: autonomyLabels[this.autonomyLevel],
      confidence: isFrench ? 'Confiance' : 'Confidence',
      acknowledgmentText: isFrench
        ? 'Je comprends que cette rponse est gnre par l\'IA et peut ncessiter une vrification.'
        : 'I understand this response is AI-generated and may require verification.'
    };
  }

  private handleAcknowledge(e: Event) {
    const checkbox = e.target as HTMLInputElement;
    this.acknowledged = checkbox.checked;

    this.dispatchEvent(new CustomEvent('acknowledgment-changed', {
      detail: { acknowledged: this.acknowledged }
    }));
  }

    Safety Guardrail Pipeline¶

    End-to-End Safety Workflow¶

    class SafetyGuardrailPipeline {
  private contentSafety: ContentSafetyService;
  private autonomyService: AutonomyLevelService;
  private hallucinationDetector: HallucinationDetector;
  private outputValidator: ChatOutputValidator;
  private auditLogger: ProtectedBAuditLogger;

  async processChatMessage(
    userId: string,
    userMessage: string,
    conversationId: string
  ): Promise<ChatResponse> {
    try {
      // Step 1: Validate user input
      const inputValidation = await this.contentSafety.analyzeText(userMessage);
      if (!inputValidation.passed) {
        return {
          success: false,
          error: inputValidation.blockedReason,
          displayMessage: 'Your message violates content policy.'
        };
      }

      // Step 2: Generate AI response with RAG
      const { response, citations } = await this.generateResponse(userMessage, conversationId);

      // Step 3: Validate AI output
      const validatedResponse = await this.outputValidator.validateAssistantResponse(response);

      // Step 4: Detect hallucinations
      const hallucinationCheck = await this.hallucinationDetector.detectHallucinations(
        validatedResponse,
        citations
      );

      if (!hallucinationCheck.isGrounded) {
        // Regenerate with stricter grounding
        return await this.regenerateWithStricterGrounding(userMessage, citations);
      }

      // Step 5: Determine autonomy level
      const autonomy = this.autonomyService.determineLevel(validatedResponse, citations);

      // Step 6: Audit log
      await this.auditLogger.logAccess({
        eventType: 'READ',
        userId,
        resourceId: conversationId,
        resourceType: 'conversation',
        classification: 'protected-b',
        ipAddress: this.getClientIP(),
        userAgent: this.getUserAgent(),
        timestamp: new Date(),
        success: true
      });

      return {
        success: true,
        message: validatedResponse,
        citations,
        autonomyLevel: autonomy.level,
        confidence: autonomy.confidence,
        disclaimer: autonomy.rationale
      };

    } catch (error) {
      // Log error and return safe fallback
      await this.auditLogger.logCriticalIncident({
        type: 'SAFETY_PIPELINE_ERROR',
        error: error.message,
        userId,
        timestamp: new Date()
      });

      return {
        success: false,
        error: 'An error occurred while processing your request.',
        displayMessage: 'Please try again or contact support.'
      };
    }
  }
}

    class SafetyMetricsService {
  trackSafetyEvent(event: SafetyEvent): void {
    this.appInsights.trackMetric({
      name: 'Safety.ContentViolations',
      value: event.blocked ? 1 : 0,
      properties: {
        severity: event.severity,
        category: event.category,
        autonomyLevel: event.autonomyLevel
      }
    });
  }

  async generateSafetyReport(startDate: Date, endDate: Date): Promise<SafetyReport> {
    const query = `
      customMetrics
      | where name == "Safety.ContentViolations"
      | where timestamp between (datetime(${startDate.toISOString()}) .. datetime(${endDate.toISOString()}))
      | summarize
          TotalChecks = count(),
          Violations = countif(value == 1),
          ViolationRate = avg(value) * 100
        by bin(timestamp, 1h)
    `;

    const results = await this.appInsights.query(query);
    return this.parseResults(results);
  }
}
