    Azure AI Foundry + MAF Migration Specification¶

    Project: EVA Autonomous Agile Crew - Azure Enterprise Migration Timeline: 30 days (Dec 11, 2025 → Jan 10, 2026) Environment: DEV - Canada Central Target: Live demonstration for stakeholder presentation Owner: Marco Presta + GitHub Copilot (SM)

    Executive Summary¶

    This specification documents the complete migration of the EVA Orchestrator autonomous task execution system from GitHub Actions to Azure AI Foundry with Microsoft Agentic Framework (MAF) and enterprise-grade orchestration.

    Current State: 3 PODs (Foundation, Operations, Quality) executing 20 production tasks autonomously via GitHub Actions, with proven parallel execution and zero race conditions.

    Target State: Enterprise Azure platform with AI Foundry model orchestration, MAF multi-agent framework, Durable Functions coordination, Cosmos DB state management, and full observability.

    Business Value: Scalable, enterprise-ready autonomous agile crew that other POs can leverage for their own delivery streams.

    # agents/base_agent.py
from maf import Agent, AgentConfig
from azure.ai.foundry import FoundryClient

class EvaBaseAgent(Agent):
    def __init__(self, agent_id: str, specialization: str):
        config = AgentConfig(
            agent_id=agent_id,
            model="gpt-4o",
            temperature=0.7,
            max_tokens=4096
        )
        super().__init__(config)
        self.specialization = specialization
        self.foundry_client = FoundryClient.from_connection_string(
            os.getenv("AZURE_AI_FOUNDRY_CONNECTION_STRING")
        )

    async def execute_task(self, task: Task) -> TaskResult:
        # Call Azure AI Foundry with task context
        response = await self.foundry_client.complete(
            model_deployment="gpt-4o",
            messages=self._build_messages(task),
            agent_context=self.specialization
        )

        # Materialize files
        files = self._extract_files(response)

        # Return structured result
        return TaskResult(
            task_id=task.id,
            status="success",
            files=files,
            summary=response.summary
        )

    2. Orchestration Logic¶

    Current: 3 separate GitHub workflow files

    Target: Single Durable Functions orchestrator

    Implementation:

    // functions/orchestrator.ts
import { app, OrchestrationContext, OrchestrationTrigger } from '@azure/functions';

export async function podOrchestrator(context: OrchestrationContext) {
    const outputs: PodResult[] = [];

    // Fan-out to 3 POD activities (parallel execution)
    const podTasks = [
        context.df.callActivity('processPodF'),
        context.df.callActivity('processPodO'),
        context.df.callActivity('processPodQ')
    ];

    // Wait for all PODs to complete
    outputs.push(...await Promise.all(podTasks));

    // Aggregate results
    const summary = {
        timestamp: new Date().toISOString(),
        pods: outputs.length,
        tasksCompleted: outputs.reduce((sum, r) => sum + r.tasksCompleted, 0),
        totalCost: outputs.reduce((sum, r) => sum + r.cost, 0)
    };

    // Write to Cosmos DB
    await context.df.callActivity('logSummary', summary);

    return summary;
}

// Register orchestrator with timer trigger
app.orchestration('PodOrchestrator', podOrchestrator);

app.timer('PodOrchestratorTimer', {
    schedule: '0 */30 * * * *', // Every 30 minutes
    handler: async (timer, context) => {
        const client = df.getClient(context);
        await client.startNew('PodOrchestrator');
    }
});

// POD Activity Functions
app.activity('processPodF', async (input: ActivityInput) => {
    // Read from Service Bus pod-f-queue
    const messages = await serviceBusClient.receiveMessages('pod-f-queue', { maxMessages: 1 });

    if (messages.length === 0) return { tasksCompleted: 0, cost: 0 };

    const task = JSON.parse(messages[0].body);

    // Invoke MAF agent
    const agent = getAgent(task.agent); // e.g., P05-SCA
    const result = await agent.executeTask(task);

    // Write result to Cosmos DB
    await cosmosClient.container('results').items.create(result);

    // Update task status
    await updateTaskStatus(task.id, 'success');

    // Complete message
    await serviceBusClient.completeMessage(messages[0]);

    return { tasksCompleted: 1, cost: result.cost };
});

// Repeat for processPodO, processPodQ

import json
from azure.cosmos import CosmosClient

def migrate_queue_to_cosmos():
    # Read JSON queue
    with open('_progress/TASK-QUEUE.json') as f:
        queue = json.load(f)

    # Connect to Cosmos DB
    client = CosmosClient.from_connection_string(conn_str)
    database = client.get_database_client('eva-orchestrator')
    container = database.get_container_client('tasks')

    # Migrate each task
    for task in queue['tasks']:
        container.upsert_item(task)

    print(f"Migrated {len(queue['tasks'])} tasks to Cosmos DB")

if __name__ == '__main__':
    migrate_queue_to_cosmos()

    Custom Metrics¶

    // functions/telemetry.ts
import { TelemetryClient } from 'applicationinsights';

export class PodTelemetry {
    private client: TelemetryClient;

    constructor() {
        this.client = new TelemetryClient(process.env.APPINSIGHTS_INSTRUMENTATIONKEY);
    }

    trackTaskExecution(pod: string, taskId: string, duration: number, cost: number) {
        this.client.trackMetric({
            name: 'TaskExecution',
            value: duration,
            properties: {
                pod,
                taskId,
                cost: cost.toString()
            }
        });
    }

    trackPodPerformance(pod: string, tasksCompleted: number, successRate: number) {
        this.client.trackMetric({
            name: 'PodPerformance',
            value: tasksCompleted,
            properties: {
                pod,
                successRate: successRate.toString()
            }
        });
    }
}
