    ğŸ”„ GitHub Actions Workflow Template¶

    Create EVA-Sovereign-UI/.github/workflows/publish-book.yml:

    name: Publish EVA-Sovereign-UI Book
# P22-DPB (Documentation Publishing Bot) - Public Book

on:
  push:
    branches:
      - main
    paths:
      - "docs/**/*.md"
      - "mkdocs.yml"
      - "docs/stylesheets/**"
      - "docs/javascripts/**"
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: "pages"
  cancel-in-progress: true

jobs:
  build:
    name: Build MkDocs Site
    runs-on: ubuntu-latest
    steps:
      - name: Checkout Repository
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"
          cache: "pip"

      - name: Install Dependencies
        run: |
          pip install --upgrade pip
          pip install mkdocs-material mkdocs-minify-plugin

      - name: Build MkDocs Site
        run: |
          mkdocs build --strict --verbose

      - name: Upload Artifact
        uses: actions/upload-pages-artifact@v3
        with:
          path: ./site

  deploy:
    name: Deploy to GitHub Pages
    needs: build
    runs-on: ubuntu-latest
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    steps:
      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4

      - name: Post-Deploy Notification
        if: success()
        run: |
          echo "âœ… EVA-Sovereign-UI Book successfully deployed!"
          echo "ğŸ“š URL: ${{ steps.deployment.outputs.page_url }}"
