    await el.updateComplete;

    expect(dismissed).to.be.true;
  });
});

          files: ./packages/web-components/coverage/lcov.info

          name: web-components

      - name: Check coverage thresholds

          cd packages/web-components

            const coverage = require('./coverage/coverage.json');
