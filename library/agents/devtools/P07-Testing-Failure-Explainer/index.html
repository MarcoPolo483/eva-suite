    Pattern P07: Testing & Failure Explainer¶

    Status: Draft Owner: POD-F (Foundation) Version: 1.0.0 Last Updated: 2025-12-12

    Overview¶

    The Testing & Failure Explainer provides test suggestions, generates failure narratives, and tracks coverage metrics to ensure 80%+ test coverage across EVA Suite.

    Purpose¶

    • Suggest test cases for uncovered code
    • Explain test failures with actionable fixes
    • Track coverage metrics and trends
    • Enforce 80% coverage quality gate

    Test Suggestions¶

    Coverage Gap Analysis¶

    Script: tools/testing/suggest-tests.ps1

    # Analyze coverage and suggest tests
param([string]$RepoPath = ".")

# Run coverage
npm run test:coverage --silent

# Parse coverage report
$coverage = Get-Content "$RepoPath/coverage/coverage-summary.json" | ConvertFrom-Json

foreach ($file in $coverage.PSObject.Properties.Name) {
    $fileCov = $coverage.$file

    if ($fileCov.lines.pct -lt 80) {
        Write-Host "`nüìù Test suggestions for: $file" -ForegroundColor Yellow
        Write-Host "  Current coverage: $($fileCov.lines.pct)%" -ForegroundColor Red

        # Suggest specific tests
        $uncoveredLines = $fileCov.lines.total - $fileCov.lines.covered
        Write-Host "  ‚ö†Ô∏è $uncoveredLines lines uncovered" -ForegroundColor Yellow

        # Parse file for uncovered functions
        $sourceCode = Get-Content $file
        Write-Host "`n  Suggested tests:"
        Write-Host "  - Test error handling paths"
        Write-Host "  - Test edge cases (null, empty, invalid input)"
        Write-Host "  - Test async/promise scenarios"
    }
}

    Auto-Generated Test Templates¶

    // tools/testing/generate-test-template.ts
export function generateTestTemplate(
  functionName: string,
  params: string[],
  returnType: string
): string {
  return `
describe('${functionName}', () => {
  it('should return ${returnType} for valid input', () => {
    const result = ${functionName}(${params.map(p => `valid${p}`).join(', ')});
    expect(result).toBeDefined();
  });

  it('should handle null input gracefully', () => {
    const result = ${functionName}(${params.map(() => 'null').join(', ')});
    expect(result).toBeNull(); // or throw error
  });

  it('should handle edge cases', () => {
    const result = ${functionName}(${params.map(() => 'edgeCase').join(', ')});
    expect(result).toMatchSnapshot();
  });
});
`.trim();
}

    Intelligent Error Parsing¶

    // tools/testing/explain-failure.ts
interface FailureNarrative {
  title: string;
  description: string;
  likelyC cause: string;
  suggestedFix: string;
  relatedDocs: string[];
}

export function explainFailure(error: string, stackTrace: string): FailureNarrative {
  // Pattern matching for common failures

  if (error.includes('Cannot read property') || error.includes('undefined')) {
    return {
      title: 'Null/Undefined Reference',
      description: 'Attempting to access property on null or undefined object',
      likelyCause: 'Missing null check or async data not loaded',
      suggestedFix: 'Add optional chaining (?.) or null guard (if/else)',
      relatedDocs: [
        'https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Operators/Optional_chaining'
      ]
    };
  }

  if (error.includes('timeout')) {
    return {
      title: 'Test Timeout',
      description: 'Test exceeded maximum execution time',
      likelyCause: 'Async operation not completing or infinite loop',
      suggestedFix: 'Increase timeout, optimize code, or fix async handling',
      relatedDocs: [
        'docs/testing/timeouts.md'
      ]
    };
  }

  if (error.includes('404') || error.includes('ECONNREFUSED')) {
    return {
      title: 'Network/API Failure',
      description: 'Unable to connect to external service',
      likelyCause: 'Service not running, wrong URL, or network issue',
      suggestedFix: 'Mock external dependencies or ensure test environment is configured',
      relatedDocs: [
        'docs/testing/mocking.md'
      ]
    };
  }

  // Generic fallback
  return {
    title: 'Test Failure',
    description: error,
    likelyCause: 'Unknown - review error message and stack trace',
    suggestedFix: 'Debug using VS Code debugger or add console.log statements',
    relatedDocs: ['docs/testing/debugging.md']
  };
}

Missing null check or async data not loaded before accessing user.name property.

### Suggested Fix
```typescript
// Before (failing):
export function getUserProfile(userId: string) {
  const user = db.users.find(userId);
  return { name: user.name, email: user.email }; // ‚ùå user might be undefined
}

// After (fixed):
export function getUserProfile(userId: string) {
  const user = db.users.find(userId);
  if (!user) {
    throw new Error(`User not found: ${userId}`);
  }
  return { name: user.name, email: user.email }; // ‚úÖ user guaranteed to exist
}

    Coverage Trends¶

    Script: tools/testing/coverage-trend.ps1

    # Track coverage over time
$reports = Get-ChildItem "docs/metrics/test-coverage-*.md" | Sort-Object Name

$trend = @()
foreach ($report in $reports) {
    $date = $report.BaseName -replace 'test-coverage-', ''
    $content = Get-Content $report.FullName -Raw

    if ($content -match 'Overall Coverage:\s+(\d+)%') {
        $coverage = [int]$matches[1]
        $trend += @{ Date = $date; Coverage = $coverage }
    }
}

# Plot trend (simplified)
Write-Host "`nCoverage Trend (last 7 days):" -ForegroundColor Cyan
foreach ($point in $trend[-7..-1]) {
    $bar = '‚ñà' * ($point.Coverage / 2)
    Write-Host "$($point.Date): $bar $($point.Coverage)%"
}
