    Pattern P10: Metrics & Insight Agent¶

    Status: Draft Owner: POD-F (Foundation) + Platform Team Version: 1.0.0 Last Updated: 2025-12-12

    Overview¶

    The Metrics & Insight Agent summarizes delivery and operational metrics, identifies trends, and suggests improvements based on data-driven insights across all EVA Suite products.

    Purpose¶

    • Collect sprint delivery metrics (velocity, burn-down, cycle time)
    • Analyze operational metrics (error rate, latency, availability)
    • Identify trends and anomalies
    • Recommend improvements based on data

    Delivery Metrics Collection¶

    Sprint Metrics Automation¶

    Script: tools/metrics/collect-sprint-metrics.ps1

    # Collect sprint delivery metrics from GitHub
param(
    [Parameter(Mandatory)]
    [string]$SprintId,

    [string]$OutputPath = "metrics/sprints"
)

Write-Host "ðŸ“Š Collecting metrics for $SprintId..." -ForegroundColor Cyan

# Get sprint milestone
$milestone = gh api "/repos/{owner}/{repo}/milestones" --jq ".[] | select(.title == \"$SprintId\")"

if (!$milestone) {
    Write-Error "Sprint milestone not found: $SprintId"
    exit 1
}

$milestoneData = $milestone | ConvertFrom-Json
$milestoneNumber = $milestoneData.number

# Get all issues in sprint
$issues = gh issue list --milestone $SprintId --state all --json number,title,state,labels,createdAt,closedAt,assignees --limit 1000 | ConvertFrom-Json

# Calculate metrics
$totalIssues = $issues.Count
$completedIssues = ($issues | Where-Object { $_.state -eq "closed" }).Count
$inProgressIssues = ($issues | Where-Object { $_.state -eq "open" -and ($_.labels.name -contains "in-progress") }).Count
$notStarted = $totalIssues - $completedIssues - $inProgressIssues

# Story points (from labels)
$totalPoints = 0
$completedPoints = 0

foreach ($issue in $issues) {
    $pointsLabel = $issue.labels | Where-Object { $_.name -match "^points-(\d+)$" }
    if ($pointsLabel) {
        $points = [int]($pointsLabel.name -replace "points-", "")
        $totalPoints += $points

        if ($issue.state -eq "closed") {
            $completedPoints += $points
        }
    }
}

# Cycle time (time from creation to completion)
$cycleTimes = @()
$completedIssues | ForEach-Object {
    if ($_.closedAt) {
        $created = [DateTime]$_.createdAt
        $closed = [DateTime]$_.closedAt
        $cycleTimes += ($closed - $created).TotalHours
    }
}

$avgCycleTime = if ($cycleTimes.Count -gt 0) {
    ($cycleTimes | Measure-Object -Average).Average
} else {
    0
}

# Generate report
$reportPath = "$OutputPath/$SprintId-metrics.json"
New-Item -Path $reportPath -ItemType File -Force | Out-Null

$report = @{
    sprint_id = $SprintId
    collection_date = (Get-Date).ToString("o")
    issues = @{
        total = $totalIssues
        completed = $completedIssues
        in_progress = $inProgressIssues
        not_started = $notStarted
        completion_rate = [math]::Round($completedIssues / $totalIssues * 100, 1)
    }
    story_points = @{
        total = $totalPoints
        completed = $completedPoints
        velocity = $completedPoints
        completion_rate = if ($totalPoints -gt 0) {
            [math]::Round($completedPoints / $totalPoints * 100, 1)
        } else { 0 }
    }
    cycle_time = @{
        avg_hours = [math]::Round($avgCycleTime, 1)
        avg_days = [math]::Round($avgCycleTime / 24, 1)
        min_hours = if ($cycleTimes.Count -gt 0) { [math]::Round(($cycleTimes | Measure-Object -Minimum).Minimum, 1) } else { 0 }
        max_hours = if ($cycleTimes.Count -gt 0) { [math]::Round(($cycleTimes | Measure-Object -Maximum).Maximum, 1) } else { 0 }
    }
    assignees = @{}
}

# Per-assignee breakdown
$issues | ForEach-Object {
    foreach ($assignee in $_.assignees) {
        if (!$report.assignees.ContainsKey($assignee.login)) {
            $report.assignees[$assignee.login] = @{
                assigned = 0
                completed = 0
            }
        }

        $report.assignees[$assignee.login].assigned++

        if ($_.state -eq "closed") {
            $report.assignees[$assignee.login].completed++
        }
    }
}

$report | ConvertTo-Json -Depth 10 | Set-Content $reportPath

# Display summary
Write-Host "`nðŸ“ˆ Sprint Summary:" -ForegroundColor Cyan
Write-Host "  Total Issues: $totalIssues" -ForegroundColor White
Write-Host "  Completed: $completedIssues ($([math]::Round($completedIssues / $totalIssues * 100, 1))%)" -ForegroundColor Green
Write-Host "  In Progress: $inProgressIssues" -ForegroundColor Yellow
Write-Host "  Not Started: $notStarted" -ForegroundColor Red
Write-Host "`nðŸ“Š Story Points:" -ForegroundColor Cyan
Write-Host "  Velocity: $completedPoints / $totalPoints ($([math]::Round($completedPoints / $totalPoints * 100, 1))%)" -ForegroundColor $(if ($completedPoints / $totalPoints -ge 0.8) { "Green" } else { "Yellow" })
Write-Host "`nâ±ï¸ Cycle Time:" -ForegroundColor Cyan
Write-Host "  Average: $([math]::Round($avgCycleTime / 24, 1)) days" -ForegroundColor White

Write-Host "`nâœ… Metrics saved: $reportPath" -ForegroundColor Green

export interface OperationalKPIs {
  // Availability
  uptime: number;              // Percentage (target: 99.9%)
  availability: number;        // Percentage (target: 99.9%)

  // Performance
  latency_p50: number;         // Milliseconds (target: <100ms)
  latency_p95: number;         // Milliseconds (target: <300ms)
  latency_p99: number;         // Milliseconds (target: <1000ms)

  // Reliability
  error_rate: number;          // Percentage (target: <1%)
  success_rate: number;        // Percentage (target: >99%)

  // Volume
  requests_per_minute: number;
  active_users: number;

  // Cost
  monthly_cost_cad: number;
  cost_per_user_cad: number;
  cost_per_request_cad: number;
}

export class KPICollector {
  async collectKPIs(appInsightsId: string, timeRange: string = '24h'): Promise<OperationalKPIs> {
    // Query Application Insights
    const queries = {
      availability: `
        requests
        | where timestamp > ago(${timeRange})
        | summarize Availability = countif(success == true) * 100.0 / count()
      `,
      latency: `
        requests
        | where timestamp > ago(${timeRange})
        | summarize
            P50 = percentile(duration, 50),
            P95 = percentile(duration, 95),
            P99 = percentile(duration, 99)
      `,
      errors: `
        requests
        | where timestamp > ago(${timeRange})
        | summarize
            ErrorRate = countif(success == false) * 100.0 / count(),
            SuccessRate = countif(success == true) * 100.0 / count()
      `,
      volume: `
        requests
        | where timestamp > ago(${timeRange})
        | summarize
            RequestsPerMin = count() / (toint(${timeRange.replace('h', '')}) * 60),
            ActiveUsers = dcount(user_Id)
      `
    };

    // Execute queries (simplified - use Azure SDK in production)
    const results = await this.executeQueries(appInsightsId, queries);

    return {
      uptime: results.availability,
      availability: results.availability,
      latency_p50: results.latency.P50,
      latency_p95: results.latency.P95,
      latency_p99: results.latency.P99,
      error_rate: results.errors.ErrorRate,

    # tools/metrics/trend_analyzer.py
"""
Analyze metric trends to identify patterns and anomalies
"""
import pandas as pd
import numpy as np
from datetime import datetime, timedelta
from typing import List, Dict, Tuple

class TrendAnalyzer:
    def __init__(self, metrics: pd.DataFrame):
        """
        Initialize with time series data

        Args:
            metrics: DataFrame with 'timestamp' and metric columns
        """
        self.metrics = metrics.set_index('timestamp')

    def detect_trend(self, column: str, window: int = 7) -> str:
        """
        Detect trend direction (increasing, decreasing, stable)

        Args:
            column: Metric column to analyze
            window: Rolling window size (days)

        Returns:
            'increasing', 'decreasing', or 'stable'
        """
        # Calculate rolling mean
        rolling_mean = self.metrics[column].rolling(window=window).mean()

        # Calculate trend slope
        recent = rolling_mean.tail(window)
        slope = (recent.iloc[-1] - recent.iloc[0]) / window

        # Determine trend direction
        std = self.metrics[column].std()
        threshold = std * 0.1  # 10% of standard deviation

        if slope > threshold:
            return 'increasing'
        elif slope < -threshold:
            return 'decreasing'
        else:
            return 'stable'

    def identify_anomalies(self, column: str, threshold: float = 3.0) -> List[Dict]:
        """
        Identify anomalies using Z-score method

        Args:
            column: Metric column to analyze
            threshold: Z-score threshold for anomaly detection

        Returns:
            List of anomalies with timestamp and value
        """
        # Calculate Z-scores
        mean = self.metrics[column].mean()
        std = self.metrics[column].std()
        z_scores = (self.metrics[column] - mean) / std

        # Find anomalies
        anomalies = []
        for timestamp, z_score in z_scores.items():
            if abs(z_score) > threshold:
                anomalies.append({
                    'timestamp': timestamp,
                    'value': self.metrics.loc[timestamp, column],
                    'z_score': z_score,
                    'severity': 'critical' if abs(z_score) > 5 else 'warning'
                })

        return anomalies

    def forecast_next_week(self, column: str) -> Tuple[float, float]:
        """
        Simple linear forecast for next week

        Args:
            column: Metric column to forecast

        Returns:
            Tuple of (forecast_value, confidence_interval)
        """
        # Use last 30 days for forecast
        recent = self.metrics[column].tail(30)

        # Linear regression
        x = np.arange(len(recent))
        y = recent.values

        # Calculate slope and intercept
        slope = np.cov(x, y)[0, 1] / np.var(x)
        intercept = y.mean() - slope * x.mean()

        # Forecast 7 days ahead
        forecast = slope * (len(recent) + 7) + intercept

        # Confidence interval (simplified)
        std = recent.std()
        confidence = 1.96 * std  # 95% confidence

        return forecast, confidence

# Example usage
if __name__ == "__main__":
    # Load metrics from JSON files
    metrics_df = pd.read_json('metrics/sprints/all-metrics.json')

    analyzer = TrendAnalyzer(metrics_df)

    # Detect velocity trend
    velocity_trend = analyzer.detect_trend('velocity')
    print(f"Velocity trend: {velocity_trend}")

    # Identify anomalies in cycle time
    cycle_time_anomalies = analyzer.identify_anomalies('avg_cycle_time_hours')
    print(f"Cycle time anomalies: {len(cycle_time_anomalies)}")

    # Forecast next sprint velocity
    next_velocity, confidence = analyzer.forecast_next_week('velocity')
    print(f"Forecast velocity: {next_velocity:.1f} Â± {confidence:.1f}")

    # tools/metrics/insight_generator.py
"""
Generate actionable insights from metrics
"""
from typing import List, Dict
from dataclasses import dataclass

@dataclass
class Insight:
    category: str  # 'performance', 'quality', 'process'
    severity: str  # 'critical', 'warning', 'info'
    title: str
    description: str
    recommendation: str
    metric_values: Dict[str, float]

class InsightGenerator:
    def __init__(self, kpis: Dict[str, float], historical: Dict[str, List[float]]):
        self.kpis = kpis
        self.historical = historical

    def generate_insights(self) -> List[Insight]:
        """Generate insights from current and historical metrics"""
        insights = []

        # Performance insights
        if self.kpis['latency_p95'] > 300:
            insights.append(Insight(
                category='performance',
                severity='warning' if self.kpis['latency_p95'] < 500 else 'critical',
                title='Elevated P95 Latency',
                description=f"P95 latency is {self.kpis['latency_p95']:.0f}ms, exceeding target of 300ms",
                recommendation="Review slow queries, optimize database indexes, consider caching strategy",
                metric_values={'p95_latency': self.kpis['latency_p95'], 'target': 300}
            ))

        # Reliability insights
        if self.kpis['error_rate'] > 1:
            insights.append(Insight(
                category='quality',
                severity='critical' if self.kpis['error_rate'] > 5 else 'warning',
                title='High Error Rate',
                description=f"Error rate is {self.kpis['error_rate']:.2f}%, exceeding target of 1%",
                recommendation="Review error logs, improve error handling, add retry logic for transient failures",
                metric_values={'error_rate': self.kpis['error_rate'], 'target': 1.0}
            ))

        # Velocity insights
        if 'velocity' in self.historical and len(self.historical['velocity']) >= 3:
            recent_velocities = self.historical['velocity'][-3:]
            avg_velocity = sum(recent_velocities) / len(recent_velocities)
            current_velocity = self.kpis.get('velocity', 0)

            if current_velocity < avg_velocity * 0.7:
                insights.append(Insight(
                    category='process',
                    severity='warning',
                    title='Declining Sprint Velocity',
                    description=f"Velocity dropped to {current_velocity}, 30% below 3-sprint average of {avg_velocity:.1f}",
                    recommendation="Review sprint planning accuracy, identify blockers, assess team capacity",
                    metric_values={'current': current_velocity, 'average': avg_velocity}
                ))

        # Cost insights
        if self.kpis.get('cost_per_user_cad', 0) > 5:
            insights.append(Insight(
                category='cost',
                severity='warning',
                title='High Cost Per User',
                description=f"Cost per user is CAD {self.kpis['cost_per_user_cad']:.2f}, exceeding target",
                recommendation="Review resource scaling, optimize compute tier, implement cost-saving strategies",
                metric_values={'cost_per_user': self.kpis['cost_per_user_cad'], 'target': 5.0}
            ))

        return insights

    def format_insights_markdown(self, insights: List[Insight]) -> str:
        """Format insights as markdown report"""
        if not insights:
            return "## âœ… All Metrics Within Target\n\nNo actionable insights at this time."

        report = "## ðŸ“Š Metric Insights\n\n"

        # Group by severity
        critical = [i for i in insights if i.severity == 'critical']
        warnings = [i for i in insights if i.severity == 'warning']
        info = [i for i in insights if i.severity == 'info']

        if critical:
            report += "### ðŸš¨ Critical Issues\n\n"
            for insight in critical:
                report += f"#### {insight.title}\n"
                report += f"**Category**: {insight.category.title()}\n\n"
                report += f"{insight.description}\n\n"
                report += f"**Recommendation**: {insight.recommendation}\n\n"

        if warnings:
            report += "### âš ï¸ Warnings\n\n"
            for insight in warnings:
                report += f"#### {insight.title}\n"
                report += f"{insight.description}\n\n"
                report += f"**Recommendation**: {insight.recommendation}\n\n"

        if info:
            report += "### â„¹ï¸ Informational\n\n"
            for insight in info:
                report += f"#### {insight.title}\n"
                report += f"{insight.description}\n\n"

        return report

# Example usage
if __name__ == "__main__":
    kpis = {
        'latency_p95': 450,
        'error_rate': 2.5,
        'velocity': 15,
        'cost_per_user_cad': 6.5
    }

    historical = {
        'velocity': [25, 23, 28]
    }

    generator = InsightGenerator(kpis, historical)
    insights = generator.generate_insights()

    print(generator.format_insights_markdown(insights))
