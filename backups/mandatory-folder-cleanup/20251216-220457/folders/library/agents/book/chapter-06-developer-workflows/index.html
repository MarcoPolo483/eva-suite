    import { CosmosClient } from "@azure/cosmos";
import { DefaultAzureCredential } from "@azure/identity";

export class CosmosDBScaler {
  private client: CosmosClient;
  private monitoringInterval: NodeJS.Timeout | null = null;
  private lastScaleTime: Date | null = null;

  constructor(private endpoint: string, private databaseId: string) {
    this.client = new CosmosClient({
      endpoint,
      aadCredentials: new DefaultAzureCredential(),
    });
  }

  /**
   * Start monitoring Cosmos DB RU consumption and auto-scale if needed
   */
  async startMonitoring(): Promise<void> {
    this.monitoringInterval = setInterval(async () => {
      await this.checkAndScale();
    }, 5 * 60 * 1000); // 5 minutes
  }

  /**
   * Check current RU consumption and scale if sustained usage > 80%
   */
  private async checkAndScale(): Promise<void> {
    try {
      const database = this.client.database(this.databaseId);
      const { resource: offer } = await database.readOffer();

      if (!offer) {
        console.warn("No offer found for database");
        return;
      }

      // Calculate RU utilization (current / max)
      const currentRU = offer.content.currentThroughput || 0;
      const maxRU = offer.content.offerThroughput || 0;
      const utilization = currentRU / maxRU;

      // Scale if utilization > 80% for 10 minutes
      if (utilization > 0.8) {
        const shouldScale = await this.checkSustainedHighUsage();
        if (shouldScale) {
          await this.scaleUp(maxRU);
        }
      }
    } catch (error) {
      console.error("Error checking Cosmos DB scaling:", error);
    }
  }

  /**
   * Check if high usage (>80%) is sustained for 10 minutes
   */
  private async checkSustainedHighUsage(): Promise<boolean> {
    // TODO: Implement 10-minute lookback window
    // For now, scale immediately if usage > 80%
    return true;
  }

  /**
   * Scale up Cosmos DB RU/s by 50%
   */
  private async scaleUp(currentRU: number): Promise<void> {
    const newRU = Math.round(currentRU * 1.5);
    console.log(`Scaling Cosmos DB from ${currentRU} RU/s to ${newRU} RU/s`);

    const database = this.client.database(this.databaseId);
    await database.replace({ throughput: newRU });

    this.lastScaleTime = new Date();
  }
}

    import { TelemetryClient } from "applicationinsights";
import { AuditTrailClient } from "@eva/provenance"; // P21 client

export class CosmosDBScaler {
  private client: CosmosClient;
  private telemetry: TelemetryClient;
  private auditTrail: AuditTrailClient;
  private monitoringInterval: NodeJS.Timeout | null = null;
  private lastScaleTime: Date | null = null;
  private cooldownPeriod = 10 * 60 * 1000; // 10 minutes

  constructor(
    private endpoint: string,
    private databaseId: string,
    telemetryClient: TelemetryClient,
    auditTrailClient: AuditTrailClient
  ) {
    this.client = new CosmosClient({
      endpoint,
      aadCredentials: new DefaultAzureCredential(),
    });
    this.telemetry = telemetryClient;
    this.auditTrail = auditTrailClient;
  }

  // ... [previous code] ...

  /**
   * Scale up Cosmos DB RU/s by 50%
   */
  private async scaleUp(currentRU: number, utilization: number): Promise<void> {
    // Check cooldown period (prevent thrashing)
    if (this.lastScaleTime) {
      const timeSinceLastScale = Date.now() - this.lastScaleTime.getTime();
      if (timeSinceLastScale < this.cooldownPeriod) {
        console.log(`Cooldown active (${Math.round(timeSinceLastScale / 1000)}s since last scale)`);
        return;
      }
    }

    const startTime = Date.now();
    const newRU = Math.round(currentRU * 1.5);

    try {
      console.log(`Scaling Cosmos DB from ${currentRU} RU/s to ${newRU} RU/s`);

      const database = this.client.database(this.databaseId);
      await database.replace({ throughput: newRU });

      const scalingDuration = Date.now() - startTime;
      this.lastScaleTime = new Date();

      // Application Insights custom event
      this.telemetry.trackEvent({
        name: "CosmosDBScalingEvent",
        properties: {
          databaseId: this.databaseId,
          oldRU: currentRU.toString(),
          newRU: newRU.toString(),
          trigger: "auto",
          cooldownMs: this.cooldownPeriod.toString(),
        },
      });

      // Application Insights metrics
      this.telemetry.trackMetric({
        name: "CosmosDB.RU.Utilization",
        value: utilization * 100,
      });

      this.telemetry.trackMetric({
        name: "CosmosDB.Scaling.Duration",
        value: scalingDuration,
      });

      // P21 audit trail (for compliance)
      await this.auditTrail.recordEvent({
        eventType: "cosmosdb.scaling.auto",
        resourceId: `cosmos://${this.endpoint}/${this.databaseId}`,
        actor: "system",
        action: "scale_up",
        metadata: {
          oldRU: currentRU,
          newRU: newRU,
          utilization: utilization,
          cooldownMs: this.cooldownPeriod,
          duration: scalingDuration,
        },
        timestamp: new Date(),
      });

      console.log(`‚úÖ Scaling complete (duration: ${scalingDuration}ms)`);
    } catch (error) {
      console.error("‚ùå Scaling failed:", error);

      // Track failure in Application Insights
      this.telemetry.trackException({
        exception: error as Error,
        properties: {
          operation: "cosmosdb.scale_up",
          databaseId: this.databaseId,
        },
      });

      throw error;
    }
  }
}

jest.mock("@azure/cosmos");

describe("CosmosDBScaler", () => {

    Troubleshooting¶

    Problem: Auto-scaling not triggering despite high utilization

    Possible Causes: 1. Cooldown period active (check Application Insights for last scaling event) 2. CosmosDBScaler service not running (check Azure Function health) 3. Managed Identity permissions missing (check Azure Key Vault access)

    Resolution: 

    # Check CosmosDBScaler service health
az functionapp show -n eva-cosmos-scaler -g eva-foundation-prod --query "state"

# Check last scaling event
az monitor app-insights query \
  --app eva-foundation-prod \
  --analytics-query "customEvents | where name == 'CosmosDBScalingEvent' | top 1 by timestamp desc"

# Restart CosmosDBScaler (if unhealthy)
az functionapp restart -n eva-cosmos-scaler -g eva-foundation-prod
