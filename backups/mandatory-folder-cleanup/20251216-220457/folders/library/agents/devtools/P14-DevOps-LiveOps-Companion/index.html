    Pattern P14: DevOps LiveOps Companion¶

    Status: Draft Owner: POD-O (Operations) Version: 1.0.0 Last Updated: 2025-12-12

    Overview¶

    The DevOps LiveOps Companion helps interpret logs, incidents, and LiveOps dashboards to support real-time operational decision-making and incident response.

    Purpose¶

    • Interpret application logs and metrics
    • Triage incidents and outages
    • Support on-call engineers during LiveOps
    • Generate incident summaries and runbook updates

    Automated Log Parsing¶

    Script: tools/liveops/analyze-logs.ps1

    # Analyze Application Insights logs for incidents
param(
    [Parameter(Mandatory)]
    [string]$AppInsightsId,

    [int]$Hours = 1
)

Write-Host "üìä Analyzing logs for last $Hours hour(s)..." -ForegroundColor Cyan

$startTime = (Get-Date).AddHours(-$Hours).ToString("o")
$endTime = (Get-Date).ToString("o")

# Query Application Insights
$query = @"
union traces, exceptions, requests
| where timestamp between (datetime($startTime) .. datetime($endTime))
| where severityLevel >= 3  // Warning and above
| summarize
    ErrorCount = countif(itemType == 'exception'),
    WarnCount = countif(severityLevel == 3),
    ErrorRate = countif(itemType == 'exception') * 100.0 / count(),
    TopErrors = make_set_if(message, itemType == 'exception', 5)
    by bin(timestamp, 5m)
| order by timestamp desc
"@

$results = az monitor app-insights query `
    --app $AppInsightsId `
    --analytics-query $query `
    --output json | ConvertFrom-Json

# Analyze patterns
$totalErrors = ($results.tables[0].rows | Measure-Object -Property 1 -Sum).Sum
$avgErrorRate = ($results.tables[0].rows | Measure-Object -Property 3 -Average).Average

Write-Host "`nIncident Summary:" -ForegroundColor Yellow
Write-Host "  Total Errors: $totalErrors" -ForegroundColor $(if ($totalErrors -gt 10) { "Red" } else { "Green" })
Write-Host "  Avg Error Rate: $([math]::Round($avgErrorRate, 2))%" -ForegroundColor $(if ($avgErrorRate -gt 5) { "Red" } else { "Green" })

# Detect anomalies
$errorSpikes = $results.tables[0].rows | Where-Object { $_[3] -gt ($avgErrorRate * 2) }

if ($errorSpikes.Count -gt 0) {
    Write-Host "`n‚ö†Ô∏è Error Rate Spikes Detected:" -ForegroundColor Red
    foreach ($spike in $errorSpikes) {
        $time = $spike[0]
        $errors = $spike[1]
        $rate = [math]::Round($spike[3], 2)
        Write-Host "  - $time : $errors errors ($rate%)" -ForegroundColor Red
    }
}

# Top errors
Write-Host "`nTop Error Messages:" -ForegroundColor Yellow
$topErrors = $results.tables[0].rows[0][4]
if ($topErrors) {
    $topErrors | ForEach-Object {
        Write-Host "  - $_" -ForegroundColor Red
    }
}

    Real-Time Anomaly Detection¶

    # tools/liveops/anomaly_detector.py
"""
Real-time anomaly detection for EVA Suite metrics
Uses statistical methods to identify abnormal patterns
"""
import pandas as pd
import numpy as np
from datetime import datetime, timedelta
from azure.monitor.query import LogsQueryClient
from azure.identity import DefaultAzureCredential

class AnomalyDetector:
    def __init__(self, app_insights_id: str):
        self.app_insights_id = app_insights_id
        self.credential = DefaultAzureCredential()
        self.client = LogsQueryClient(self.credential)

    def detect_error_rate_anomalies(self, hours: int = 24) -> list:
        """Detect error rate anomalies using Z-score"""
        # Query error rates
        query = f"""
        requests
        | where timestamp > ago({hours}h)
        | summarize
            TotalRequests = count(),
            FailedRequests = countif(success == false),
            ErrorRate = countif(success == false) * 100.0 / count()
            by bin(timestamp, 5m)
        | order by timestamp asc
        """

        result = self.client.query_workspace(
            workspace_id=self.app_insights_id,
            query=query,
            timespan=timedelta(hours=hours)
        )

        # Convert to DataFrame
        df = pd.DataFrame(result.tables[0].rows, columns=[c.name for c in result.tables[0].columns])

        # Calculate Z-scores
        df['ErrorRate_ZScore'] = (df['ErrorRate'] - df['ErrorRate'].mean()) / df['ErrorRate'].std()

        # Flag anomalies (|Z| > 3)
        anomalies = df[abs(df['ErrorRate_ZScore']) > 3]

        return anomalies.to_dict('records')

    def detect_latency_anomalies(self, percentile: float = 0.95) -> list:
        """Detect latency anomalies using percentile thresholds"""
        query = f"""
        requests
        | where timestamp > ago(24h)
        | summarize
            P50 = percentile(duration, 50),
            P95 = percentile(duration, 95),
            P99 = percentile(duration, 99)
            by bin(timestamp, 5m)
        | order by timestamp asc
        """

        result = self.client.query_workspace(
            workspace_id=self.app_insights_id,
            query=query,
            timespan=timedelta(hours=24)
        )

        df = pd.DataFrame(result.tables[0].rows, columns=[c.name for c in result.tables[0].columns])

        # Detect when P95 exceeds 3x median P95
        threshold = df['P95'].median() * 3
        anomalies = df[df['P95'] > threshold]

        return anomalies.to_dict('records')

    def generate_incident_summary(self, anomalies: list) -> str:
        """Generate human-readable incident summary"""
        if not anomalies:
            return "No anomalies detected."

        summary = f"## Incident Summary\n\n"
        summary += f"**Time Range**: Last 24 hours\n"
        summary += f"**Anomalies Detected**: {len(anomalies)}\n\n"

        for i, anomaly in enumerate(anomalies, 1):
            summary += f"### Anomaly {i}\n"
            summary += f"- **Time**: {anomaly['timestamp']}\n"
            summary += f"- **Error Rate**: {anomaly['ErrorRate']:.2f}%\n"
            summary += f"- **Z-Score**: {anomaly['ErrorRate_ZScore']:.2f}\n"
            summary += f"- **Severity**: {'Critical' if abs(anomaly['ErrorRate_ZScore']) > 5 else 'Warning'}\n\n"

        return summary

# Example usage
if __name__ == "__main__":
    detector = AnomalyDetector(app_insights_id="<workspace-id>")

    error_anomalies = detector.detect_error_rate_anomalies(hours=24)
    latency_anomalies = detector.detect_latency_anomalies()

    print(detector.generate_incident_summary(error_anomalies))

on:
  workflow_dispatch:
    inputs:
      incident_id:
        description: 'Incident ID'
        required: true
      severity:
        description: 'Severity (P0/P1/P2/P3)'
        required: true

        description: 'Brief incident description'
        required: true

jobs:
  triage:
    runs-on: ubuntu-latest
    steps:
      - name: Create incident ticket
        run: |
          gh issue create \
            --title "INCIDENT-${{ github.event.inputs.incident_id }}: ${{ github.event.inputs.description }}" \
            --label "incident,priority-${{ github.event.inputs.severity }}" \
            --body "## Incident Details

          **ID**: ${{ github.event.inputs.incident_id }}
          **Severity**: ${{ github.event.inputs.severity }}
          **Reported**: $(date -u)
          **Description**: ${{ github.event.inputs.description }}

          ## Triage Checklist

          - [ ] Impact assessment completed
          - [ ] Root cause identified
          - [ ] Mitigation deployed
          - [ ] Customers notified
          - [ ] Postmortem scheduled

          ## Timeline

          - **T+0**: Incident detected
          "

      - name: Notify on-call team
        run: |
          # Send Teams notification (placeholder)
          curl -X POST ${{ secrets.TEAMS_WEBHOOK }} \
            -H 'Content-Type: application/json' \
            -d '{
              "title": "üö® Incident ${{ github.event.inputs.incident_id }}",
              "text": "${{ github.event.inputs.description }}",
              "themeColor": "FF0000"
            }'

      - name: Collect diagnostic logs
        run: |
          # Query Application Insights for last hour
          az monitor app-insights query \
            --app ${{ secrets.APP_INSIGHTS_ID }} \
            --analytics-query "traces | where timestamp > ago(1h) | where severityLevel >= 3" \
            --output json > diagnostic-logs.json

      - name: Run anomaly detection
        run: |
          python tools/liveops/anomaly_detector.py \
            --app-insights-id ${{ secrets.APP_INSIGHTS_ID }} \
            --hours 2 > anomaly-report.md

      - name: Attach diagnostics to incident
        run: |
          # Upload logs to incident ticket
          gh issue comment ${{ steps.create-issue.outputs.issue-number }} \
            --body-file anomaly-report.md

    Automated Incident Summarization¶

    # tools/liveops/incident_summarizer.py
"""
Generate natural language incident summaries from metrics
"""
from dataclasses import dataclass
from datetime import datetime
from typing import List

@dataclass
class MetricAnomaly:
    metric_name: str
    timestamp: datetime
    value: float
    baseline: float
    severity: str  # 'warning' or 'critical'

class IncidentSummarizer:
    def summarize(self, anomalies: List[MetricAnomaly]) -> str:
        """Generate natural language incident summary"""
        if not anomalies:
            return "No incidents detected. All systems operational."

        # Group by severity
        critical = [a for a in anomalies if a.severity == 'critical']
        warnings = [a for a in anomalies if a.severity == 'warning']

        summary = "## Incident Summary\n\n"

        if critical:
            summary += f"### üö® Critical Issues ({len(critical)})\n\n"
            for anomaly in critical:
                deviation = ((anomaly.value - anomaly.baseline) / anomaly.baseline) * 100
                summary += f"- **{anomaly.metric_name}** is {abs(deviation):.1f}% {'above' if deviation > 0 else 'below'} baseline "
                summary += f"({anomaly.value:.2f} vs {anomaly.baseline:.2f})\n"
                summary += f"  - First detected: {anomaly.timestamp.strftime('%Y-%m-%d %H:%M:%S UTC')}\n"

        if warnings:
            summary += f"\n### ‚ö†Ô∏è Warnings ({len(warnings)})\n\n"
            for anomaly in warnings:
                deviation = ((anomaly.value - anomaly.baseline) / anomaly.baseline) * 100
                summary += f"- **{anomaly.metric_name}** is {abs(deviation):.1f}% {'above' if deviation > 0 else 'below'} baseline\n"

        # Suggest actions
        summary += "\n## Recommended Actions\n\n"

        if any(a.metric_name == 'errorRate' for a in critical):
            summary += "1. Check application logs for error patterns\n"
            summary += "2. Verify recent deployments or configuration changes\n"
            summary += "3. Review database connection health\n"

        if any(a.metric_name == 'latencyP95' for a in critical):
            summary += "1. Check downstream service dependencies\n"
            summary += "2. Verify database query performance\n"
            summary += "3. Review cache hit rates\n"

        if any(a.metric_name == 'availability' for a in critical):
            summary += "1. Check service health endpoints\n"
            summary += "2. Verify infrastructure availability (VMs, Load Balancers)\n"
            summary += "3. Review recent Azure service health notifications\n"

        return summary
