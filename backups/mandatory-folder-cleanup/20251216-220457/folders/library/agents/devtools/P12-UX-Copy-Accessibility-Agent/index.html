    Pattern P12: UX, Copy & Accessibility Agent¶

    Status: Draft Owner: POD-X (UX/UI) Version: 1.0.0 Last Updated: 2025-12-12

    Overview¶

    The UX, Copy & Accessibility Agent validates UX flows, ensures WCAG 2.2 AAA compliance, and reviews bilingual copy (EN-CA/FR-CA) to maintain GC Design System standards across all EVA Suite products.

    Purpose¶

    • Validate UX flows and user journeys
    • Ensure WCAG 2.2 AAA accessibility compliance
    • Review bilingual copy for accuracy and consistency
    • Enforce GC Design System alignment

    Accessibility Validation¶

    axe-core Integration¶

    Script: tools/ux/accessibility-scan.ps1

    # Run accessibility scan on UI applications
param(
    [string[]]$Apps = @("eva-sovereign-ui", "eva-admin-ui")
)

foreach ($app in $Apps) {
    Write-Host "`n♿ Scanning: $app" -ForegroundColor Cyan

    Push-Location "../$app"

    # Run axe-core via test suite
    npm run test:a11y

    # Parse results
    $results = Get-Content "a11y-report.json" | ConvertFrom-Json

    $violations = $results.violations.Count
    $passes = $results.passes.Count

    if ($violations -gt 0) {
        Write-Host "  ❌ $violations accessibility violations found" -ForegroundColor Red

        foreach ($violation in $results.violations) {
            Write-Host "    - $($violation.id): $($violation.description)" -ForegroundColor Red
            Write-Host "      Impact: $($violation.impact)" -ForegroundColor Yellow
            Write-Host "      Nodes affected: $($violation.nodes.Count)" -ForegroundColor Yellow
        }
    } else {
        Write-Host "  ✅ No violations ($passes checks passed)" -ForegroundColor Green
    }

    Pop-Location
}

    WCAG 2.2 AAA Checklist¶

    wcag_compliance:
  perceivable:
    - Text alternatives for non-text content (1.1.1)
    - Captions for audio/video (1.2.1-1.2.9)
    - Adaptable layouts (1.3.1-1.3.6)
    - Color contrast ≥7:1 for normal text (1.4.6 AAA)
    - Color contrast ≥4.5:1 for large text (1.4.6 AAA)
    - No loss of content when zoomed to 200% (1.4.4)

  operable:
    - Keyboard accessible (2.1.1-2.1.4)
    - No keyboard traps (2.1.2)
    - Sufficient time for tasks (2.2.1-2.2.6)
    - No flashing content >3 times/second (2.3.1-2.3.3)
    - Multiple ways to navigate (2.4.5)
    - Focus visible (2.4.7)

  understandable:
    - Page language identified (3.1.1)
    - Language of parts identified (3.1.2)
    - Consistent navigation (3.2.3)
    - Consistent identification (3.2.4)
    - Error suggestions provided (3.3.3)

  robust:
    - Valid HTML/ARIA (4.1.1-4.1.3)
    - Name, role, value for UI components (4.1.2)

    Bilingual Copy Review¶

    Review Process¶

    Workflow: 1. Extract all user-facing text from UI components 2. Verify EN-CA translation exists 3. Verify FR-CA translation exists 4. Check consistency with GC terminology database 5. Flag missing or inconsistent translations

    Script: tools/ux/bilingual-check.ps1

    # Check bilingual completeness
param([string]$AppPath = ".")

Write-Host "🌐 Checking bilingual compliance..." -ForegroundColor Cyan

# Find all i18n files
$enFiles = Get-ChildItem -Path "$AppPath/src/locales/en-CA" -Filter "*.json" -Recurse
$frFiles = Get-ChildItem -Path "$AppPath/src/locales/fr-CA" -Filter "*.json" -Recurse

# Compare EN and FR files
$enCount = $enFiles.Count
$frCount = $frFiles.Count

if ($enCount -ne $frCount) {
    Write-Host "  ⚠️ File count mismatch: $enCount EN files, $frCount FR files" -ForegroundColor Yellow
}

foreach ($enFile in $enFiles) {
    $relativePath = $enFile.FullName.Replace("$AppPath/src/locales/en-CA/", "")
    $frFile = "$AppPath/src/locales/fr-CA/$relativePath"

    if (!(Test-Path $frFile)) {
        Write-Host "  ❌ Missing FR translation: $relativePath" -ForegroundColor Red
        continue
    }

    # Load both files
    $enContent = Get-Content $enFile.FullName | ConvertFrom-Json
    $frContent = Get-Content $frFile | ConvertFrom-Json

    # Compare keys
    $enKeys = ($enContent | Get-Member -MemberType NoteProperty).Name
    $frKeys = ($frContent | Get-Member -MemberType NoteProperty).Name

    $missingKeys = $enKeys | Where-Object { $_ -notin $frKeys }

    if ($missingKeys.Count -gt 0) {
        Write-Host "  ⚠️ Missing FR keys in $relativePath" -ForegroundColor Yellow
        foreach ($key in $missingKeys) {
            Write-Host "    - $key" -ForegroundColor Yellow
        }
    } else {
        Write-Host "  ✅ $relativePath complete" -ForegroundColor Green
    }
}

    GC Terminology Validation¶

    Reference: Government of Canada Translation Bureau TERMIUM Plus

    // tools/ux/validate-gc-terms.ts
const gcTerminology = {
  // Common GC terms with official translations
  'Government of Canada': 'Gouvernement du Canada',
  'My Account': 'Mon dossier',
  'Sign In': 'Ouvrir une session',
  'Sign Out': 'Fermer la session',
  'Help': 'Aide',
  'Contact Us': 'Contactez-nous',
  // ... more terms from TERMIUM Plus
};

export function validateTerminology(enText: string, frText: string): string[] {
  const issues: string[] = [];

  for (const [en, fr] of Object.entries(gcTerminology)) {
    if (enText.includes(en) && !frText.includes(fr)) {
      issues.push(`Expected FR term "${fr}" for EN term "${en}"`);
    }
  }

  return issues;
}

    GC Design System Compliance¶

    Component Validation¶

    // tools/ux/validate-gc-components.ts
interface ComponentAudit {
  component: string;
  gcCompliant: boolean;
  issues: string[];
}

export function auditGcCompliance(component: string): ComponentAudit {
  const issues: string[] = [];

  // Check for GC Design System tokens
  if (!component.includes('var(--gc-')) {
    issues.push('Not using GC Design System CSS custom properties');
  }

  // Check for canada-gc theme
  if (!component.includes('canada-gc')) {
    issues.push('Missing canada-gc theme class');
  }

  // Check for bilingual attributes
  if (!component.includes('lang=') && component.includes('<html')) {
    issues.push('Missing lang attribute on HTML element');
  }

  return {
    component,
    gcCompliant: issues.length === 0,
    issues
  };
}
