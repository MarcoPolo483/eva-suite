    Pattern P15: Dev Master Orchestrator (Jarvis)¶

    Status: Draft Owner: POD-F (Foundation) Version: 1.0.0 Last Updated: 2025-12-12

    Overview¶

    The Dev Master Orchestrator (Jarvis) coordinates all other DevTools agents (P01-P14) across multiple repositories, orchestrating multi-step workflows and ensuring consistent execution of development patterns.

    Purpose¶

    • Orchestrate multi-agent workflows across repos
    • Coordinate P01-P14 agent invocations
    • Ensure consistent pattern execution
    • Track cross-repo dependencies and impacts

    Orchestrator Architecture¶

    Workflow Orchestration¶

    Script: tools/jarvis/orchestrator.ps1

    # Jarvis Orchestrator - Multi-agent workflow coordination
param(
    [Parameter(Mandatory)]
    [string]$WorkflowName,

    [hashtable]$Context = @{}
)

Write-Host " Jarvis Orchestrator Starting..." -ForegroundColor Cyan
Write-Host "Workflow: $WorkflowName" -ForegroundColor Yellow

# Load workflow definition
$workflowPath = "tools/jarvis/workflows/$WorkflowName.yaml"
if (!(Test-Path $workflowPath)) {
    Write-Error "Workflow not found: $WorkflowName"
    exit 1
}

$workflow = Get-Content $workflowPath | ConvertFrom-Yaml

Write-Host "`nWorkflow Steps: $($workflow.steps.Count)" -ForegroundColor Cyan

# Execute workflow steps
$results = @{}

foreach ($step in $workflow.steps) {
    Write-Host "`n Step $($step.id): $($step.name)" -ForegroundColor Green

    # Resolve agent from registry
    $agent = Get-Agent -PatternCode $step.agent

    if (!$agent) {
        Write-Error "Agent not found: $($step.agent)"
        exit 1
    }

    # Build parameters from context
    $params = @{}
    foreach ($key in $step.params.Keys) {
        $value = $step.params[$key]

        # Resolve references to previous step outputs
        if ($value -match '^\$\{(.+)\}$') {
            $reference = $Matches[1]
            $value = $results[$reference]
        }

        $params[$key] = $value
    }

    # Execute agent
    Write-Host "  Invoking: $($agent.name)" -ForegroundColor Yellow
    $output = & $agent.endpoint @params

    # Store result
    $results[$step.id] = $output

    # Check for errors
    if ($LASTEXITCODE -ne 0) {
        Write-Error "Step $($step.id) failed with exit code $LASTEXITCODE"

        if ($step.on_error -eq 'fail') {
            exit 1
        } elseif ($step.on_error -eq 'continue') {
            Write-Warning "Continuing despite error (on_error=continue)"
        }
    }

    Write-Host "   Step $($step.id) completed" -ForegroundColor Green
}

Write-Host "`n Workflow '$WorkflowName' completed successfully!" -ForegroundColor Green

# Return final result
return $results

    Cross-Repo Impact Analysis¶

    # tools/jarvis/impact-analysis.ps1
param(
    [Parameter(Mandatory)]
    [string]$TargetRepo,

    [Parameter(Mandatory)]
    [string]$ChangeType  # 'breaking', 'feature', 'fix'
)

Write-Host " Analyzing impact of $ChangeType change to $TargetRepo..." -ForegroundColor Cyan

# Load repo graph
$graph = Get-Content "tools/jarvis/repo-graph.json" | ConvertFrom-Json

$repo = $graph.repositories.$TargetRepo

if (!$repo) {
    Write-Error "Repository not found: $TargetRepo"
    exit 1
}

# Identify impacted repos
$impacted = @()

if ($ChangeType -eq 'breaking') {
    # Breaking changes impact all dependents
    $impacted += $repo.dependents

    Write-Host "`n Breaking change detected!" -ForegroundColor Red
    Write-Host "Directly impacted repositories: $($repo.dependents.Count)" -ForegroundColor Yellow

    foreach ($dependent in $repo.dependents) {
        Write-Host "  - $dependent" -ForegroundColor Red

        # Check transitive dependents
        $transitive = $graph.repositories.$dependent.dependents
        if ($transitive) {
            Write-Host "    Transitive: $($transitive -join ', ')" -ForegroundColor Yellow
            $impacted += $transitive
        }
    }
}

# Generate impact report
$reportPath = "reports/impact-$TargetRepo-$(Get-Date -Format 'yyyyMMdd-HHmmss').md"
New-Item -Path $reportPath -ItemType File -Force | Out-Null

@"
# Impact Analysis: $TargetRepo

**Change Type**: $ChangeType
**Date**: $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss')

## Directly Impacted Repositories

$($repo.dependents | ForEach-Object { "- $_" } | Out-String)

## Required Actions

$( if ($ChangeType -eq 'breaking') {
    @"
###  Breaking Change Protocol

1. **Notify** all dependent repository owners
2. **Update** migration guide in docs/
3. **Schedule** deprecation timeline (minimum 2 sprints)
4. **Coordinate** synchronized releases

### Dependent Repo Checklist

$($repo.dependents | ForEach-Object {
    @"
#### $_
- [ ] Team notified
- [ ] Migration plan created
- [ ] Tests updated
- [ ] Documentation updated
- [ ] Release scheduled
"@
} | Out-String)
"@
} else {
    "No breaking changes. Standard release process applies."
})

## Recommended Workflow

\`\`\`powershell
# Execute coordinated deployment
tools/jarvis/orchestrator.ps1 -WorkflowName coordinated-release -Context @{
    source_repo = "$TargetRepo"
    change_type = "$ChangeType"
    impacted_repos = @($($repo.dependents | ForEach-Object { "'$_'" } -join ', '))
}
\`\`\`
"@ | Set-Content $reportPath

Write-Host "`n Impact analysis complete: $reportPath" -ForegroundColor Green
