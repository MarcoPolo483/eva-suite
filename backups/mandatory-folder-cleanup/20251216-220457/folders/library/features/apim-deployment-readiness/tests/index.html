  And 101st request fails (HTTP 429)
  And Retry-After header present

    Test Steps: 

    # 1. Send 101 requests rapidly
$subKey = az apim subscription show --resource-group rg-evada2 `
  --service-name eva-apim-demo --subscription-id marco-demo-001 `
  --query "primaryKey" -o tsv

$results = @()
for ($i=1; $i -le 101; $i++) {
    $response = curl https://eva-apim-demo.azure-api.net/eva/health `
      -H "Ocp-Apim-Subscription-Key: $subKey" `
      -s -o $null -w "%{http_code}"
    $results += $response
    if ($i -le 5 -or $i -ge 98) {
        Write-Host "Request $i: HTTP $response"
    }
}

# 2. Count successes (200) and rate limits (429)
$successes = ($results | Where-Object { $_ -eq "200" }).Count
$rateLimits = ($results | Where-Object { $_ -eq "429" }).Count

Write-Host "Successes: $successes, Rate Limits: $rateLimits"

# 3. Verify rate limiting
if ($successes -eq 100 -and $rateLimits -eq 1) {
    Write-Host "‚úÖ PASS: Rate limiting enforced correctly" -ForegroundColor Green

    Write-Host "‚ùå FAIL: Rate limiting incorrect" -ForegroundColor Red
}

# 4. Check Retry-After header

  -H "Ocp-Apim-Subscription-Key: $subKey" -i | Select-String "Retry-After"

    T-FR5-2: Rate Limit Headers¶

    Priority: P0 Requirement: FR-5 Type: Functional

    Scenario: 

    Scenario: Rate limit headers present in responses
  Given user makes 1st request
  When checking response headers
  Then X-RateLimit-Limit equals 100
  And X-RateLimit-Remaining equals 99
  And X-RateLimit-Reset is future timestamp

    Test Steps: 

    # 1. Make request and capture headers
$response = curl https://eva-apim-demo.azure-api.net/eva/health `
  -H "Ocp-Apim-Subscription-Key: $subKey" -i

# 2. Extract rate limit headers
$limit = $response | Select-String "X-RateLimit-Limit: (\d+)" | % { $_.Matches.Groups[1].Value }
$remaining = $response | Select-String "X-RateLimit-Remaining: (\d+)" | % { $_.Matches.Groups[1].Value }
$reset = $response | Select-String "X-RateLimit-Reset: (\d+)" | % { $_.Matches.Groups[1].Value }

Write-Host "Limit: $limit, Remaining: $remaining, Reset: $reset"

# 3. Verify headers
$allPresent = $limit -and $remaining -and $reset
if ($allPresent -and $limit -eq "100" -and $remaining -le $limit) {
    Write-Host "‚úÖ PASS: Rate limit headers correct" -ForegroundColor Green
} else {
    Write-Host "‚ùå FAIL: Rate limit headers missing or incorrect" -ForegroundColor Red
}
