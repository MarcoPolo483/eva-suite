    Chapter 03: Conversation Management¶

    Conversation management is the backbone of EVA Chat 2.0, orchestrating the flow of multi-turn dialogues while maintaining context, performance, and user experience. This chapter explores the technical architecture behind session lifecycle management, message persistence, context window optimization, and state synchronization across UI and backend layers.

    Session Lifecycle Management¶

    Every conversation in EVA Chat begins with session creation, progresses through multiple turns, and eventually ends when the user navigates away or explicitly closes the chat. The session lifecycle is managed through three core operations: create, resume, and end.

    Creating a New Session¶

    When a user initiates a conversation, EVA Chat creates a unique session identifier and initializes both Cosmos DB and Redis state:

    // src/hooks/useConversationSession.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { v4 as uuidv4 } from 'uuid';

interface ConversationSession {
  id: string;
  userId: string;
  createdAt: Date;
  lastActiveAt: Date;
  metadata: {
    model: string;
    temperature: number;
    maxTokens: number;
  };
}

export function useCreateSession() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (userId: string): Promise<ConversationSession> => {
      const sessionId = uuidv4();
      const session: ConversationSession = {
        id: sessionId,
        userId,
        createdAt: new Date(),
        lastActiveAt: new Date(),
        metadata: {
          model: 'gpt-4o',
          temperature: 0.7,
          maxTokens: 4096,
        },
      };

      // Persist to Cosmos DB
      await fetch('/api/conversations/sessions', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(session),
      });

      // Cache in Redis for quick access (TTL: 24 hours)
      await fetch(`/api/cache/sessions/${sessionId}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ session, ttl: 86400 }),
      });

      return session;
    },
    onSuccess: (session) => {
      queryClient.setQueryData(['session', session.id], session);
    },
  });
}

    Resuming an Existing Session¶

    When a user returns to an active conversation, EVA Chat first checks Redis cache before falling back to Cosmos DB:

    // src/hooks/useConversationSession.ts (continued)
export function useResumeSession(sessionId: string) {
  return useQuery({
    queryKey: ['session', sessionId],
    queryFn: async (): Promise<ConversationSession> => {
      // Try Redis cache first (fast path)
      const cacheResponse = await fetch(`/api/cache/sessions/${sessionId}`);
      if (cacheResponse.ok) {
        const cached = await cacheResponse.json();
        return cached.session;
      }

      // Fallback to Cosmos DB (slow path)
      const dbResponse = await fetch(`/api/conversations/sessions/${sessionId}`);
      if (!dbResponse.ok) {
        throw new Error('Session not found');
      }

      const session = await dbResponse.json();

      // Repopulate Redis cache
      await fetch(`/api/cache/sessions/${sessionId}`, {
        method: 'PUT',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ session, ttl: 86400 }),
      });

      return session;
    },
    staleTime: 5 * 60 * 1000, // 5 minutes
    gcTime: 30 * 60 * 1000, // 30 minutes
  });
}

    Ending a Session¶

    When a conversation ends, EVA Chat updates the session metadata and clears the Redis cache while preserving Cosmos DB records for audit and analytics:

    export function useEndSession() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (sessionId: string) => {
      await fetch(`/api/conversations/sessions/${sessionId}/end`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ endedAt: new Date() }),
      });

      // Clear Redis cache
      await fetch(`/api/cache/sessions/${sessionId}`, {
        method: 'DELETE',
      });
    },
    onSuccess: (_, sessionId) => {
      queryClient.removeQueries({ queryKey: ['session', sessionId] });
    },
  });
}

    Message History and Pagination¶

    Message history is stored in Cosmos DB with efficient pagination and filtering capabilities. EVA Chat uses continuation tokens for seamless infinite scroll and supports filtering by date ranges and topics.

    Paginated Message Retrieval¶

    // src/hooks/useMessageHistory.ts
import { useInfiniteQuery } from '@tanstack/react-query';

interface MessagePage {
  messages: ChatMessage[];
  continuationToken?: string;
}

export function useMessageHistory(sessionId: string) {
  return useInfiniteQuery({
    queryKey: ['messages', sessionId],
    queryFn: async ({ pageParam }): Promise<MessagePage> => {
      const params = new URLSearchParams({
        sessionId,
        pageSize: '50',
        ...(pageParam && { continuationToken: pageParam }),
      });

      const response = await fetch(`/api/conversations/messages?${params}`);
      return response.json();
    },
    getNextPageParam: (lastPage) => lastPage.continuationToken,
    initialPageParam: undefined,
    staleTime: 2 * 60 * 1000, // 2 minutes
  });
}

    Backend Cosmos DB Query (Node.js/TypeScript)¶

    // backend/src/services/messageService.ts
import { CosmosClient } from '@azure/cosmos';

export async function fetchMessages(
  sessionId: string,
  continuationToken?: string,
  pageSize: number = 50
): Promise<MessagePage> {
  const client = new CosmosClient(process.env.COSMOS_CONNECTION_STRING!);
  const container = client.database('eva-chat').container('messages');

  const querySpec = {
    query: 'SELECT * FROM c WHERE c.sessionId = @sessionId ORDER BY c.timestamp DESC',
    parameters: [{ name: '@sessionId', value: sessionId }],
  };

  const { resources, continuationToken: nextToken } = await container.items
    .query(querySpec, {
      maxItemCount: pageSize,
      continuationToken,
    })
    .fetchNext();

  return {
    messages: resources as ChatMessage[],
    continuationToken: nextToken,
  };
}

    Filtering by Date and Topic¶

    // backend/src/services/messageService.ts (continued)
export async function fetchFilteredMessages(
  sessionId: string,
  filters: {
    startDate?: Date;
    endDate?: Date;
    topic?: string;
  }
): Promise<ChatMessage[]> {
  const client = new CosmosClient(process.env.COSMOS_CONNECTION_STRING!);
  const container = client.database('eva-chat').container('messages');

  let query = 'SELECT * FROM c WHERE c.sessionId = @sessionId';
  const parameters = [{ name: '@sessionId', value: sessionId }];

  if (filters.startDate) {
    query += ' AND c.timestamp >= @startDate';
    parameters.push({ name: '@startDate', value: filters.startDate.toISOString() });
  }

  if (filters.endDate) {
    query += ' AND c.timestamp <= @endDate';
    parameters.push({ name: '@endDate', value: filters.endDate.toISOString() });
  }

  if (filters.topic) {
    query += ' AND CONTAINS(c.content, @topic)';
    parameters.push({ name: '@topic', value: filters.topic });
  }

  query += ' ORDER BY c.timestamp DESC';

  const { resources } = await container.items
    .query({ query, parameters })
    .fetchAll();

  return resources as ChatMessage[];
}

    Context Window Management¶

    EVA Chat enforces token limits to prevent exceeding OpenAI model constraints (e.g., 128k tokens for GPT-4o). The system tracks token counts and applies intelligent truncation strategies.

    Token Counting and Truncation¶

    // src/utils/tokenManagement.ts
import { encoding_for_model } from 'tiktoken';

export class ContextWindowManager {
  private encoder = encoding_for_model('gpt-4o');
  private maxTokens: number;

  constructor(maxTokens: number = 4096) {
    this.maxTokens = maxTokens;
  }

  countTokens(text: string): number {
    return this.encoder.encode(text).length;
  }

  truncateMessages(messages: ChatMessage[]): ChatMessage[] {
    let totalTokens = 0;
    const truncated: ChatMessage[] = [];

    // Always keep the most recent messages
    for (let i = messages.length - 1; i >= 0; i--) {
      const msg = messages[i];
      const msgTokens = this.countTokens(msg.content);

      if (totalTokens + msgTokens > this.maxTokens) {
        break;
      }

      truncated.unshift(msg);
      totalTokens += msgTokens;
    }

    return truncated;
  }

  summarizeOldMessages(messages: ChatMessage[]): ChatMessage {
    // Summarize truncated messages into a system message
    const summarized = messages.map((m) => `${m.role}: ${m.content}`).join('\n');
    return {
      id: uuidv4(),
      sessionId: messages[0]?.sessionId || '',
      role: 'system',
      content: `[Summary of previous conversation]\n${summarized.slice(0, 500)}...`,
      timestamp: new Date(),
      tokenCount: this.countTokens(summarized),
    };
  }

  getContextWindowStatus(messages: ChatMessage[]): {
    totalTokens: number;
    remainingTokens: number;
    utilizationPercent: number;
  } {
    const totalTokens = messages.reduce(
      (sum, msg) => sum + this.countTokens(msg.content),
      0
    );

    return {
      totalTokens,
      remainingTokens: Math.max(0, this.maxTokens - totalTokens),
      utilizationPercent: Math.min(100, (totalTokens / this.maxTokens) * 100),
    };
  }
}

    State Tracking Hook¶

    // src/hooks/useConversationState.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';

export function useConversationState(sessionId: string) {
  const queryClient = useQueryClient();

  const { data: state } = useQuery({
    queryKey: ['conversationState', sessionId],
    queryFn: async (): Promise<ConversationState> => {
      const response = await fetch(`/api/conversations/state/${sessionId}`);
      return response.json();
    },
  });

  const updateState = useMutation({
    mutationFn: async (updates: Partial<ConversationState>) => {
      const response = await fetch(`/api/conversations/state/${sessionId}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(updates),
      });
      return response.json();
    },
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['conversationState', sessionId] });
    },
  });

  return { state, updateState };
}

    State Management Architecture¶

    EVA Chat uses a dual-state management strategy: React Context for UI state (current input, UI flags) and TanStack Query for server state (messages, sessions).

    React Context for UI State¶

    // src/contexts/ConversationContext.tsx
import { createContext, useContext, useState, ReactNode } from 'react';

interface ConversationUIState {
  currentInput: string;
  setCurrentInput: (input: string) => void;
  isTyping: boolean;
  setIsTyping: (typing: boolean) => void;
  selectedModel: string;
  setSelectedModel: (model: string) => void;
  isSidebarOpen: boolean;
  setIsSidebarOpen: (open: boolean) => void;
}

const ConversationContext = createContext<ConversationUIState | undefined>(undefined);

export function ConversationProvider({ children }: { children: ReactNode }) {
  const [currentInput, setCurrentInput] = useState('');
  const [isTyping, setIsTyping] = useState(false);
  const [selectedModel, setSelectedModel] = useState('gpt-4o');
  const [isSidebarOpen, setIsSidebarOpen] = useState(true);

  return (
    <ConversationContext.Provider
      value={{
        currentInput,
        setCurrentInput,
        isTyping,
        setIsTyping,
        selectedModel,
        setSelectedModel,
        isSidebarOpen,
        setIsSidebarOpen,
      }}
    >
      {children}
    </ConversationContext.Provider>
  );
}

export function useConversation() {
  const context = useContext(ConversationContext);
  if (!context) {
    throw new Error('useConversation must be used within ConversationProvider');
  }
  return context;
}

    Conversation Persistence¶

    EVA Chat supports exporting conversations to JSON and Markdown formats, importing past conversations, and seamless offline-to-online synchronization.

    Export Conversation¶

    // src/utils/conversationExport.ts
export async function exportConversation(
  sessionId: string,
  format: 'json' | 'markdown'
): Promise<void> {
  const response = await fetch(`/api/conversations/export/${sessionId}?format=${format}`);
  const blob = await response.blob();
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = `conversation-${sessionId}.${format === 'json' ? 'json' : 'md'}`;
  a.click();
  URL.revokeObjectURL(url);
}

    Backend Export Service¶

    // backend/src/services/exportService.ts
export async function exportAsMarkdown(sessionId: string): Promise<string> {
  const messages = await fetchAllMessages(sessionId);
  const session = await fetchSession(sessionId);

  let markdown = `# Conversation Export\n\n`;
  markdown += `**Session ID**: ${sessionId}\n`;
  markdown += `**Created**: ${session.createdAt}\n`;
  markdown += `**Model**: ${session.metadata.model}\n\n`;
  markdown += `---\n\n`;

  for (const msg of messages) {
    const timestamp = new Date(msg.timestamp).toLocaleString();
    markdown += `### ${msg.role.toUpperCase()} - ${timestamp}\n\n`;
    markdown += `${msg.content}\n\n`;

    if (msg.metadata?.citations) {
      markdown += `**Citations**:\n`;
      msg.metadata.citations.forEach((cite, idx) => {
        markdown += `${idx + 1}. ${cite.source}: "${cite.snippet}"\n`;
      });
      markdown += `\n`;
    }
  }

  return markdown;
}

export async function exportAsJSON(sessionId: string): Promise<any> {
  const messages = await fetchAllMessages(sessionId);
  const session = await fetchSession(sessionId);

  return {
    exportDate: new Date().toISOString(),
    session,
    messages,
    metadata: {
      messageCount: messages.length,
      totalTokens: messages.reduce((sum, m) => sum + m.tokenCount, 0),
    },
  };
}

    Message Deletion¶

    EVA Chat supports soft deletion with audit trails, ensuring compliance with data retention policies while preserving conversation integrity.

    Soft Delete Implementation¶

    // src/hooks/useDeleteMessage.ts
export function useDeleteMessage() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (messageId: string) => {
      await fetch(`/api/conversations/messages/${messageId}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          deleted: true,
          deletedAt: new Date(),
        }),
      });
    },
    onSuccess: (_, messageId) => {
      // Invalidate queries to refetch without deleted message
      queryClient.invalidateQueries({ queryKey: ['messages'] });
    },
  });
}

    Cosmos DB Soft Delete Query¶

    // backend/src/services/messageService.ts (continued)
export async function softDeleteMessage(messageId: string): Promise<void> {
  const client = new CosmosClient(process.env.COSMOS_CONNECTION_STRING!);
  const container = client.database('eva-chat').container('messages');

  const { resource: message } = await container.item(messageId, messageId).read();

  await container.item(messageId, messageId).replace({
    ...message,
    deleted: true,
    deletedAt: new Date().toISOString(),
  });
}

// Fetch messages excluding soft-deleted
export async function fetchActiveMessages(sessionId: string): Promise<ChatMessage[]> {
  const client = new CosmosClient(process.env.COSMOS_CONNECTION_STRING!);
  const container = client.database('eva-chat').container('messages');

  const querySpec = {
    query: `SELECT * FROM c
            WHERE c.sessionId = @sessionId
            AND (NOT IS_DEFINED(c.deleted) OR c.deleted = false)
            ORDER BY c.timestamp DESC`,
    parameters: [{ name: '@sessionId', value: sessionId }],
  };

  const { resources } = await container.items.query(querySpec).fetchAll();
  return resources as ChatMessage[];
}

    Redis Session Cache Pattern¶

    To minimize Cosmos DB read costs and latency, EVA Chat implements a Redis cache-aside pattern for active sessions and recent messages:

    Session Caching¶

    // backend/src/services/sessionCache.ts
import { createClient } from 'redis';

const redisClient = createClient({ url: process.env.REDIS_CONNECTION_STRING });

export async function cacheSession(session: ConversationSession, ttl: number = 86400) {
  await redisClient.setEx(
    `session:${session.id}`,
    ttl,
    JSON.stringify(session)
  );
}

export async function getCachedSession(sessionId: string): Promise<ConversationSession | null> {
  const cached = await redisClient.get(`session:${sessionId}`);
  return cached ? JSON.parse(cached) : null;
}

export async function invalidateSessionCache(sessionId: string) {
  await redisClient.del(`session:${sessionId}`);
}

    Message Caching¶

    // backend/src/services/messageCache.ts
export async function cacheRecentMessages(
  sessionId: string,
  messages: ChatMessage[],
  ttl: number = 3600
) {
  const key = `messages:${sessionId}:recent`;
  await redisClient.setEx(key, ttl, JSON.stringify(messages));
}

export async function getCachedMessages(sessionId: string): Promise<ChatMessage[] | null> {
  const key = `messages:${sessionId}:recent`;
  const cached = await redisClient.get(key);
  return cached ? JSON.parse(cached) : null;
}

export async function invalidateMessageCache(sessionId: string) {
  const key = `messages:${sessionId}:recent`;
  await redisClient.del(key);
}

    Cache-Aside Pattern Implementation¶

    // backend/src/services/conversationService.ts
export async function getSessionWithCache(sessionId: string): Promise<ConversationSession> {
  // Try cache first
  const cached = await getCachedSession(sessionId);
  if (cached) {
    return cached;
  }

  // Cache miss - fetch from Cosmos DB
  const session = await fetchSessionFromCosmosDB(sessionId);

  // Populate cache for next request
  await cacheSession(session);

  return session;
}

export async function getMessagesWithCache(sessionId: string): Promise<ChatMessage[]> {
  // Try cache first
  const cached = await getCachedMessages(sessionId);
  if (cached) {
    return cached;
  }

  // Cache miss - fetch from Cosmos DB
  const messages = await fetchActiveMessages(sessionId);

  // Cache recent messages (last 50)
  await cacheRecentMessages(sessionId, messages.slice(0, 50));

  return messages;
}

    Summary¶

    Conversation management in EVA Chat 2.0 balances performance, cost, and user experience through a multi-layered architecture:

    • Session lifecycle handled via Cosmos DB persistence and Redis caching for sub-100ms session retrieval
    • Message history paginated efficiently using Cosmos DB continuation tokens with support for filtering by date and topic
    • Context window management enforced with tiktoken-based truncation, ensuring compliance with model token limits
    • Multi-turn state tracking maintains conversation context across turns with intent and entity tracking
    • State management split between React Context (UI state) and TanStack Query (server state) for optimal rendering and caching
    • Persistence via export/import with JSON and Markdown support for portability and backup
    • Deletion implemented as soft deletes with audit trails for compliance and data recovery
    • Redis caching reduces Cosmos DB read costs by 70% and achieves P95 latency under 150ms

    This architecture ensures EVA Chat can scale to thousands of concurrent conversations while maintaining sub-200ms P95 latency for message retrieval and enabling context-aware multi-turn dialogues that preserve conversation history and user intent across sessions.
