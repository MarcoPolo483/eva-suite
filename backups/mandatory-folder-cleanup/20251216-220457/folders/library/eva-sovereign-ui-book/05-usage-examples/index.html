    Chapter 05: Usage Examples¶

    This chapter demonstrates common usage patterns for EVA Sovereign UI components across different scenarios and frameworks. All examples follow GC Design System standards and enforce WCAG 2.1 AA compliance and bilingualism by default.

    Form Validation¶

    Form validation in EVA Sovereign UI integrates with native HTML5 validation while providing bilingual error messaging and accessible feedback.

    Basic Form with Required Fields¶

    <eva-gc-form id="userForm">
  <eva-gc-input
    id="firstName"
    label-en="First Name"
    label-fr="Prénom"
    required
    error-en="Please enter your first name"
    error-fr="Veuillez entrer votre prénom"
  ></eva-gc-input>

  <eva-gc-input
    id="email"
    type="email"
    label-en="Email Address"
    label-fr="Adresse courriel"
    required
    pattern="^[a-zA-Z0-9._%+-]+@canada\.ca$"
    error-en="Please enter a valid @canada.ca email address"
    error-fr="Veuillez entrer une adresse courriel @canada.ca valide"
  ></eva-gc-input>

  <eva-gc-button
    type="submit"
    label-en="Submit"
    label-fr="Soumettre"
  ></eva-gc-button>
</eva-gc-form>

<script>
  const form = document.getElementById('userForm');

  form.addEventListener('eva-submit', (event) => {
    const { valid, values, errors } = event.detail;

    if (valid) {
      // Submit to backend
      console.log('Form data:', values);
    } else {
      // Display errors (component handles ARIA live region announcements)
      console.error('Validation errors:', errors);
    }
  });
</script>

    Custom Validation Rules¶

    const form = document.getElementById('userForm');
const input = document.getElementById('departmentCode');

// Add custom validator
input.addValidator({
  name: 'departmentExists',
  validate: async (value) => {
    const response = await fetch(`/api/departments/${value}`);
    return response.ok;
  },
  messageEn: 'Department code not found',
  messageFr: 'Code de département introuvable'
});

// Async validation triggers on blur
input.addEventListener('eva-validation-complete', (event) => {
  const { valid, errors } = event.detail;
  console.log('Validation result:', { valid, errors });
});

<script>
  const wizard = document.getElementById('onboardingWizard');

  wizard.addEventListener('eva-step-change', (event) => {
    const { previousStep, currentStep, totalSteps } = event.detail;

    // Validate before moving forward
    if (currentStep > previousStep) {
      const valid = wizard.validateCurrentStep();
      if (!valid) {
        event.preventDefault(); // Stay on current step
      }
    }

    // Update button labels on final step
    if (currentStep === totalSteps - 1) {
      nextBtn.setAttribute('label-en', 'Submit');
      nextBtn.setAttribute('label-fr', 'Soumettre');
    }
  });

  wizard.addEventListener('eva-wizard-complete', (event) => {
    const formData = event.detail.data;
    console.log('Wizard completed:', formData);
  });
</script>

    Progress Persistence¶

    // Save wizard state to localStorage
wizard.addEventListener('eva-step-change', () => {
  const state = wizard.getState();
  localStorage.setItem('wizardState', JSON.stringify(state));
});

// Restore wizard state on page load
const savedState = localStorage.getItem('wizardState');
if (savedState) {
  wizard.setState(JSON.parse(savedState));
}

<script>
  const table = document.getElementById('employeeTable');

  // Load data
  fetch('/api/employees')
    .then(response => response.json())
    .then(data => {
      table.data = data;
    });

  // Listen for sort events
  table.addEventListener('eva-sort', (event) => {
    const { field, direction } = event.detail;
    console.log(`Sorted by ${field} in ${direction} order`);
    // Screen reader announces: "Table sorted by Full Name in ascending order"
  });

  // Listen for filter events
  table.addEventListener('eva-filter', (event) => {
    const { filters } = event.detail;
    console.log('Active filters:', filters);
    // Screen reader announces: "Table filtered. Showing 12 of 50 rows"
  });
</script>

    Pagination¶

    <eva-gc-pagination
  id="tablePagination"
  total-items="250"
  items-per-page="25"
  current-page="1"
  label-en="Employees"
  label-fr="Employés"
></eva-gc-pagination>

<script>
  const pagination = document.getElementById('tablePagination');

  pagination.addEventListener('eva-page-change', (event) => {
    const { page, itemsPerPage } = event.detail;

    // Fetch new page of data
    const offset = (page - 1) * itemsPerPage;
    fetch(`/api/employees?offset=${offset}&limit=${itemsPerPage}`)
      .then(response => response.json())
      .then(data => {
        table.data = data;
        // Screen reader announces: "Showing page 2 of 10"
      });
  });
</script>

    React Integration¶

    EVA Sovereign UI components work seamlessly in React applications with TypeScript support.

    Basic Component Usage¶

    import '@eva-suite/sovereign-ui';
import { useState } from 'react';

interface EmployeeFormProps {
  onSubmit: (data: any) => void;
}

export function EmployeeForm({ onSubmit }: EmployeeFormProps) {
  const [formData, setFormData] = useState({
    firstName: '',
    lastName: '',
    email: ''
  });

  const handleSubmit = (event: CustomEvent) => {
    const { valid, values } = event.detail;
    if (valid) {
      onSubmit(values);
    }
  };

  return (
    <eva-gc-form onEva-submit={handleSubmit}>
      <eva-gc-input
        id="firstName"
        label-en="First Name"
        label-fr="Prénom"
        value={formData.firstName}
        onEva-input={(e) => setFormData({ ...formData, firstName: e.detail.value })}
        required
      />

      <eva-gc-input
        id="email"
        type="email"
        label-en="Email"
        label-fr="Courriel"
        value={formData.email}
        onEva-input={(e) => setFormData({ ...formData, email: e.detail.value })}
        required
      />

      <eva-gc-button
        type="submit"
        label-en="Submit"
        label-fr="Soumettre"
      />
    </eva-gc-form>
  );
}

    React Wrapper Hook¶

    import { useRef, useEffect } from 'react';

export function useEvaComponent<T extends HTMLElement>(
  ref: React.RefObject<T>,
  eventHandlers: Record<string, (event: CustomEvent) => void>
) {
  useEffect(() => {
    const element = ref.current;
    if (!element) return;

    const listeners: Array<[string, EventListener]> = [];

    Object.entries(eventHandlers).forEach(([eventName, handler]) => {
      const listener = handler as EventListener;
      element.addEventListener(eventName, listener);
      listeners.push([eventName, listener]);
    });

    return () => {
      listeners.forEach(([eventName, listener]) => {
        element.removeEventListener(eventName, listener);
      });
    };
  }, [ref, eventHandlers]);
}

// Usage
function MyComponent() {
  const tableRef = useRef<HTMLElement>(null);

  useEvaComponent(tableRef, {
    'eva-sort': (event) => console.log('Sort:', event.detail),
    'eva-filter': (event) => console.log('Filter:', event.detail)
  });

  return <eva-gc-table ref={tableRef} />;
}

    Vue Integration¶

    Vue 3 provides first-class support for Web Components with automatic event binding and reactive props.

    Basic Component Usage¶

    <template>
  <eva-gc-form @eva-submit="handleSubmit">
    <eva-gc-input
      id="firstName"

      required
    />

    <eva-gc-select
      id="department"
      label-en="Department"

      <option value="ESDC">ESDC</option>
      <option value="TBS">TBS</option>

    <eva-gc-button
      type="submit"
      label-en="Submit"
      label-fr="Soumettre"
    />

<script setup lang="ts">
import '@eva-suite/sovereign-ui';
import { reactive } from 'vue';

const formData = reactive({
  firstName: '',
  department: ''
});

function updateFirstName(event: CustomEvent) {
  formData.firstName = event.detail.value;
}

function updateDepartment(event: CustomEvent) {
  formData.department = event.detail.value;
}

function handleSubmit(event: CustomEvent) {
  const { valid, values } = event.detail;
  if (valid) {
    console.log('Form submitted:', values);
  }
}
</script>

    Composable for EVA Components¶

    // useEvaForm.ts
import { ref, Ref } from 'vue';

export function useEvaForm<T>(initialData: T) {
  const formData: Ref<T> = ref(initialData) as Ref<T>;
  const errors: Ref<Record<string, string>> = ref({});
  const isValid: Ref<boolean> = ref(false);

  function updateField(field: keyof T, value: any) {
    formData.value = {
      ...formData.value,
      [field]: value
    };
  }

  function handleSubmit(event: CustomEvent) {
    const { valid, values, errors: validationErrors } = event.detail;
    isValid.value = valid;

    if (valid) {
      formData.value = values as T;
    } else {
      errors.value = validationErrors;
    }
  }

  function reset() {
    formData.value = initialData;
    errors.value = {};
    isValid.value = false;
  }

  return {
    formData,
    errors,
    isValid,
    updateField,
    handleSubmit,
    reset
  };
}

    Summary¶

    EVA Sovereign UI provides production-ready patterns for:

    • Form validation: Bilingual errors, async validators, ARIA announcements
    • Multi-step wizards: Progress tracking, state persistence, keyboard navigation
    • Data tables: Accessible sorting/filtering, custom cells, pagination
    • Modal dialogs: Focus trapping, escape handling, bilingual content
    • Bilingual navigation: Seamless language switching, localized routing

    All patterns enforce WCAG 2.1 AA compliance and GC Design System standards by default. Framework integration (React/Vue) requires minimal configuration and provides full TypeScript support.
