    // eva-suite/src/components/RouteFocus.tsx
import { useEffect, useRef } from 'react';
import { useLocation } from 'react-router-dom';

export function RouteFocus({ children }: { children: React.ReactNode }) {
  const location = useLocation();
  const mainRef = useRef<HTMLElement>(null);

  useEffect(() => {
    // Focus main region on route change
    mainRef.current?.focus();

    // Announce route change to screen readers
    announce(`Navigated to ${document.title}`, 'polite');
  }, [location.pathname]);

  return (
    <main ref={mainRef} id="main" tabIndex={-1}>
      {children}
    </main>
  );
}

    // eva-suite/src/services/analytics/pii-tracker.ts
export interface PIIEvent {
  timestamp: Date;
  sessionId: string;
  userId?: string;
  piiType: 'email' | 'phone' | 'sin' | 'address' | 'health' | 'financial';
  detected: boolean;
  redacted: boolean;
  context: 'chat' | 'form' | 'upload';
  action: 'stored' | 'processed' | 'transmitted';
}

export function trackPII(event: PIIEvent): void {
  // Log to GA4 as custom event
  gtag('event', 'pii_detection', {
    pii_type: event.piiType,
    context: event.context,
    redacted: event.redacted,
    action: event.action,
    non_interaction: true, // Don't affect bounce rate
  });

  // Log to Azure Monitor (audit trail)
  logger.audit('pii_detection', event);

  // Alert if PII not redacted
  if (event.detected && !event.redacted) {
    alertService.critical('PII detected without redaction', event);
  }
}

    4. Model Drift Detection ‚úÖ¶

    Implementation: 

    // eva-suite/src/services/analytics/model-drift.ts
export interface ModelPerformanceMetrics {
  modelId: string;
  period: 'daily' | 'weekly' | 'monthly';
  metrics: {
    acceptanceRate: number;        // % of recommendations accepted
    averageConfidence: number;      // Mean confidence score
    errorRate: number;              // % of errors/hallucinations
    latency: number;                // P95 response time
    userSatisfaction: number;       // Explicit ratings
  };
  baseline: ModelPerformanceMetrics['metrics'];
  drift: {
    acceptanceRate: number;         // % change from baseline
    confidence: number;
    errorRate: number;
    latency: number;
    userSatisfaction: number;
  };
  alert: boolean;                   // Drift >10% triggers alert
}

export function detectModelDrift(
  current: ModelPerformanceMetrics['metrics'],
  baseline: ModelPerformanceMetrics['metrics']
): boolean {
  const thresholds = {
    acceptanceRate: 0.10,   // 10% drop = alert
    errorRate: 0.10,         // 10% increase = alert
    latency: 0.20,           // 20% increase = alert
    userSatisfaction: 0.15,  // 15% drop = alert
  };

  const drifts = {
    acceptanceRate: (baseline.acceptanceRate - current.acceptanceRate) / baseline.acceptanceRate,
    errorRate: (current.errorRate - baseline.errorRate) / baseline.errorRate,
    latency: (current.latency - baseline.latency) / baseline.latency,
    userSatisfaction: (baseline.userSatisfaction - current.userSatisfaction) / baseline.userSatisfaction,
  };

  // Alert if any metric exceeds threshold
  return Object.entries(drifts).some(([metric, drift]) => {
    return Math.abs(drift) > thresholds[metric as keyof typeof thresholds];
  });
}
