    Hero Demo Launcher Component Specification¶

    Component: DemoLauncher Purpose: Portal component that launches Azure-hosted hero demos via iframe or new tab Location: eva-suite/src/components/DemoLauncher.tsx Related: REQ-EVA-Suite-Portal.md STORY-003, AZURE-HERO-DEMO-DEPLOYMENT.md

    Component Interface¶

    interface DemoLauncherProps {
  demoId: string;                // e.g., "eva-da", "eva-chat"
  launchMode?: "iframe" | "newtab"; // Default from registry
  showHeader?: boolean;          // Show demo metadata header (default: true)
  showGraduation?: boolean;      // Show graduation checklist (default: false)
}

interface HeroDemo {
  id: string;                    // e.g., "eva-da"
  name: string;                  // e.g., "EVA Domain Assistant"
  nameI18n: { en: string; fr: string };
  description: string;
  descriptionI18n: { en: string; fr: string };
  azureUrl: string;              // e.g., "https://eva-da-dev.z9.web.core.windows.net/"
  launchMode: "iframe" | "newtab";
  status: "hero-demo" | "alpha" | "beta" | "production";
  graduationTarget: string;      // e.g., "Q1 2026"
  owner: string;                 // POD assignment
  tags: string[];
  screenshot: string;            // URL to screenshot
  graduationCriteria: {
    repoCreated: boolean;
    prodDeployed: boolean;
    bookPublished: boolean;
    heroCardUpdated: boolean;
  };
  azureResources: {
    devStaticWebApp: string;     // e.g., "eva-da-dev-swa"
    alphaStaticWebApp?: string;
    betaAppService?: string;
    prodAppService?: string;
  };
}

    Component Implementation¶

    File: eva-suite/src/components/DemoLauncher.tsx

    import React, { useEffect } from 'react';
import { useParams, Navigate } from 'react-router-dom';
import { HERO_DEMOS } from '../data/hero-demos';
import { DemoHeader } from './DemoHeader';
import { GraduationChecklist } from './GraduationChecklist';
import './DemoLauncher.css';

interface DemoLauncherProps {
  demoId?: string;
  launchMode?: "iframe" | "newtab";
  showHeader?: boolean;
  showGraduation?: boolean;
}

export function DemoLauncher({
  demoId: propDemoId,
  launchMode: propLaunchMode,
  showHeader = true,
  showGraduation = false
}: DemoLauncherProps) {
  const params = useParams();
  const demoId = propDemoId || params.demoId;

  if (!demoId) {
    return <Navigate to="/demos" replace />;
  }

  const demo = HERO_DEMOS[demoId];

  if (!demo) {
    return (
      <div className="demo-launcher-error">
        <h1>Demo Not Found</h1>
        <p>Demo "{demoId}" does not exist.</p>
        <a href="/demos">‚Üê Back to Hero Demos</a>
      </div>
    );
  }

  const launchMode = propLaunchMode || demo.launchMode;

  // New Tab mode: Redirect immediately
  useEffect(() => {
    if (launchMode === "newtab") {
      window.location.href = demo.azureUrl;
    }
  }, [launchMode, demo.azureUrl]);

  if (launchMode === "newtab") {
    return (
      <div className="demo-launcher-redirect">
        <div className="spinner" aria-label="Loading demo..." />
        <p>Launching {demo.name}...</p>
        <p className="text-muted">
          <a href={demo.azureUrl} target="_blank" rel="noopener noreferrer">
            Click here if not redirected
          </a>
        </p>
      </div>
    );
  }

  // Iframe mode: Embed demo
  return (
    <div className="demo-launcher-container">
      {showHeader && <DemoHeader demo={demo} />}

      <div className="demo-iframe-wrapper">
        <iframe
          src={demo.azureUrl}
          title={demo.name}
          className="demo-iframe"
          sandbox="allow-scripts allow-same-origin allow-forms allow-popups"
          referrerPolicy="strict-origin-when-cross-origin"
          loading="lazy"
        />
      </div>

      {showGraduation && <GraduationChecklist demo={demo} />}
    </div>
  );
}

    GraduationChecklist Component¶

    File: eva-suite/src/components/GraduationChecklist.tsx

    import React from 'react';
import { HeroDemo } from '../data/hero-demos';
import { useTranslation } from 'react-i18next';
import './GraduationChecklist.css';

interface GraduationChecklistProps {
  demo: HeroDemo;
}

export function GraduationChecklist({ demo }: GraduationChecklistProps) {
  const { t } = useTranslation();
  const criteria = demo.graduationCriteria;

  const allComplete = Object.values(criteria).every(v => v === true);
  const completedCount = Object.values(criteria).filter(v => v === true).length;
  const totalCount = Object.keys(criteria).length;

  return (
    <aside className="graduation-checklist">
      <h2>{t('demo.graduationChecklist')}</h2>
      <p className="checklist-progress">
        {completedCount} / {totalCount} {t('demo.criteriaComplete')}
      </p>

      <ul className="checklist-items">
        <li className={criteria.repoCreated ? 'complete' : 'incomplete'}>
          <input
            type="checkbox"
            checked={criteria.repoCreated}
            disabled
            aria-label={t('demo.criteria.repoCreated')}
          />

        </li>
        <li className={criteria.prodDeployed ? 'complete' : 'incomplete'}>

            type="checkbox"

    Acceptance Criteria¶

    • DemoLauncher component renders for all 6 hero demos
    • New tab mode redirects immediately to Azure URL
    • Iframe mode embeds Azure demo with proper sandbox attributes
    • Demo header displays metadata (name, description, owner, graduation target)
    • Graduation checklist shows 4 criteria with checkboxes
    • Bilingual support (EN-CA/FR-CA) via i18n
    • WCAG 2.1 AA compliant (keyboard nav, ARIA labels, focus indicators)
    • Responsive design (mobile, tablet, desktop)
    • Loading spinner shown during new tab redirect
    • Error handling for invalid demo IDs (404 page)
