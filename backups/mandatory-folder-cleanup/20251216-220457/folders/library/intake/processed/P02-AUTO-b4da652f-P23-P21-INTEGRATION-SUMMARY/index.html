    New Functions Added (57 lines): 

    # SHA256 fingerprinting (P21 integration)
function Get-FileFingerprint {
    param([string]$FilePath)
    $hash = Get-FileHash -Path $FilePath -Algorithm SHA256
    return @{
        path = $FilePath
        sha256 = $hash.Hash
        size = (Get-Item $FilePath).Length
        lastModified = (Get-Item $FilePath).LastWriteTime.ToString("o")
    }
}

# Provenance metadata generator (P21 integration)
function New-ProvenanceRecord {
    param(
        [string]$Action,      # "deleted", "moved", "archived"
        [string]$SourcePath,
        [string]$DestPath = $null,
        [string]$Reason,
        [string]$Category,
        [hashtable]$Fingerprint
    )
    return @{
        timestamp = (Get-Date).ToUniversalTime().ToString("o")
        action = $Action
        sourcePath = $SourcePath
        destinationPath = $DestPath
        reason = $Reason
        category = $Category
        fingerprint = $Fingerprint
        executedBy = $env:USERNAME
        machine = $env:COMPUTERNAME
        runId = $script:runId
    }
}
