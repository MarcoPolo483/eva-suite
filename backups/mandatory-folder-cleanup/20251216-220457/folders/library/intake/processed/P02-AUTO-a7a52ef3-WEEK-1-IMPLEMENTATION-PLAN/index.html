    import { LitElement, html, css } from 'lit';
import { customElement, property, state } from 'lit/decorators.js';
import { classMap } from 'lit/directives/class-map.js';

/**
 * GC-compliant government header with wordmark, language toggle, and navigation
 * @element eva-gov-header
 *
 * @attr {string} site-title - The title of the website
 * @attr {string} site-title-fr - French title (bilingual)
 * @attr {string} current-lang - Current language ('en' or 'fr')
 * @attr {boolean} show-search - Show search functionality
 * @attr {boolean} show-breadcrumbs - Show breadcrumb navigation
 *
 * @fires language-change - Fired when language toggle clicked
 * @fires search-submit - Fired when search submitted
 *
 * @slot navigation - Main navigation menu items
 * @slot breadcrumbs - Breadcrumb trail
 */
@customElement('eva-gov-header')
export class EvaGovHeader extends LitElement {
  static styles = css`
    :host {
      display: block;
      background: #fff;
      border-bottom: 4px solid #26374a;
    }

    .header-container {
      max-width: 1200px;
      margin: 0 auto;
      padding: 1rem;
    }

    .top-bar {
      display: flex;
      justify-content: space-between;
      align-items: center;
      margin-bottom: 1rem;
    }

    .gc-wordmark {
      display: flex;
      align-items: center;
      gap: 1rem;
    }

    .gc-wordmark img {
      height: 40px;
    }

    .site-title {
      font-size: 1.5rem;
      font-weight: 600;
      color: #26374a;
      margin: 0;
    }

    .language-toggle {
      background: transparent;
      border: none;
      color: #26374a;
      text-decoration: underline;
      cursor: pointer;
      font-size: 1rem;
      padding: 0.5rem 1rem;
    }

    .language-toggle:hover {
      color: #0535d2;
    }

    .search-container {
      margin: 1rem 0;
    }

    .search-form {
      display: flex;
      gap: 0.5rem;
      max-width: 600px;
    }

    .search-input {
      flex: 1;
      padding: 0.5rem;
      border: 2px solid #26374a;
      font-size: 1rem;
    }

    .search-button {
      padding: 0.5rem 1.5rem;
      background: #26374a;
      color: white;
      border: none;
      cursor: pointer;
      font-size: 1rem;
    }

    .search-button:hover {
      background: #0535d2;
    }

    @media (max-width: 768px) {
      .top-bar {
        flex-direction: column;
        align-items: flex-start;
        gap: 1rem;
      }

      .search-form {
        flex-direction: column;
      }
    }
  `;

  @property({ type: String, attribute: 'site-title' })
  siteTitle = '';

  @property({ type: String, attribute: 'site-title-fr' })
  siteTitleFr = '';

  @property({ type: String, attribute: 'current-lang' })
  currentLang: 'en' | 'fr' = 'en';

  @property({ type: Boolean, attribute: 'show-search' })
  showSearch = true;

  @property({ type: Boolean, attribute: 'show-breadcrumbs' })
  showBreadcrumbs = false;

  @state()
  private searchQuery = '';

  private handleLanguageToggle() {
    const newLang = this.currentLang === 'en' ? 'fr' : 'en';
    this.dispatchEvent(new CustomEvent('language-change', {
      detail: { lang: newLang },
      bubbles: true,
      composed: true
    }));
  }

  private handleSearchSubmit(e: Event) {
    e.preventDefault();
    this.dispatchEvent(new CustomEvent('search-submit', {
      detail: { query: this.searchQuery },
      bubbles: true,
      composed: true
    }));
  }

  render() {
    const displayTitle = this.currentLang === 'en' ? this.siteTitle : this.siteTitleFr;

    return html`
      <header class="header-container">
        <div class="top-bar">
          <div class="gc-wordmark">
            <img
              src="https://www.canada.ca/etc/designs/canada/wet-boew/assets/sig-blk-en.svg"
              alt="${this.currentLang === 'en' ? 'Government of Canada' : 'Gouvernement du Canada'}"
            />
            ${displayTitle ? html`<h1 class="site-title">${displayTitle}</h1>` : ''}
          </div>

          <button
            class="language-toggle"
            @click=${this.handleLanguageToggle}
            aria-label="${this.currentLang === 'en' ? 'FranÃ§ais' : 'English'}"
          >
            ${this.currentLang === 'en' ? 'FranÃ§ais' : 'English'}
          </button>
        </div>

        ${this.showSearch ? html`
          <div class="search-container">
            <form class="search-form" @submit=${this.handleSearchSubmit}>
              <input
                type="text"
                class="search-input"
                placeholder="${this.currentLang === 'en' ? 'Search Canada.ca' : 'Rechercher dans Canada.ca'}"
                .value=${this.searchQuery}
                @input=${(e: Event) => this.searchQuery = (e.target as HTMLInputElement).value}
                aria-label="${this.currentLang === 'en' ? 'Search' : 'Rechercher'}"
              />
              <button type="submit" class="search-button">
                ${this.currentLang === 'en' ? 'Search' : 'Rechercher'}
              </button>
            </form>
          </div>
        ` : ''}

        ${this.showBreadcrumbs ? html`
          <nav aria-label="${this.currentLang === 'en' ? 'Breadcrumb' : 'Fil d\'Ariane'}">
            <slot name="breadcrumbs"></slot>
          </nav>
        ` : ''}

        <nav aria-label="${this.currentLang === 'en' ? 'Main navigation' : 'Navigation principale'}">
          <slot name="navigation"></slot>
        </nav>
      </header>
    `;
  }
}

    import { describe, it, expect, beforeEach, vi } from 'vitest';
import { fixture, html } from '@open-wc/testing';
import './eva-gov-header';
import type { EvaGovHeader } from './eva-gov-header';

describe('eva-gov-header', () => {
  let element: EvaGovHeader;

  beforeEach(async () => {
    element = await fixture(html`<eva-gov-header></eva-gov-header>`);
  });

  it('should render with default properties', async () => {
    expect(element).toBeTruthy();
    expect(element.currentLang).toBe('en');
    expect(element.showSearch).toBe(true);
    expect(element.showBreadcrumbs).toBe(false);
  });

  it('should display GC wordmark', async () => {
    const img = element.shadowRoot?.querySelector('.gc-wordmark img');
    expect(img).toBeTruthy();
    expect(img?.getAttribute('alt')).toContain('Government of Canada');
  });

  it('should render site title when provided', async () => {
    element.siteTitle = 'EVA Suite';
    await element.updateComplete;

    const title = element.shadowRoot?.querySelector('.site-title');
    expect(title?.textContent).toBe('EVA Suite');
  });

  it('should toggle language', async () => {
    const langChangeSpy = vi.fn();
    element.addEventListener('language-change', langChangeSpy);

    const toggleBtn = element.shadowRoot?.querySelector('.language-toggle') as HTMLButtonElement;
    toggleBtn.click();
    await element.updateComplete;

    expect(langChangeSpy).toHaveBeenCalledWith(
      expect.objectContaining({
        detail: { lang: 'fr' }
      })
    );
  });

  it('should handle search submission', async () => {
    const searchSpy = vi.fn();
    element.addEventListener('search-submit', searchSpy);

    const input = element.shadowRoot?.querySelector('.search-input') as HTMLInputElement;
    const form = element.shadowRoot?.querySelector('.search-form') as HTMLFormElement;

    input.value = 'test query';
    input.dispatchEvent(new Event('input'));
    await element.updateComplete;

    form.submit();
    await element.updateComplete;

    expect(searchSpy).toHaveBeenCalledWith(
      expect.objectContaining({
        detail: { query: 'test query' }
      })
    );
  });

  it('should hide search when showSearch is false', async () => {
    element.showSearch = false;
    await element.updateComplete;

    const searchContainer = element.shadowRoot?.querySelector('.search-container');
    expect(searchContainer).toBeNull();
  });

  it('should display breadcrumbs slot when enabled', async () => {
    element.showBreadcrumbs = true;
    await element.updateComplete;

    const breadcrumbNav = element.shadowRoot?.querySelector('nav[aria-label*="Breadcrumb"]');
    expect(breadcrumbNav).toBeTruthy();
  });

  // Accessibility tests
  it('should have proper ARIA labels', async () => {
    const toggleBtn = element.shadowRoot?.querySelector('.language-toggle');

    expect(searchInput?.getAttribute('aria-label')).toBeTruthy();

  });

    expect(title?.textContent).toBe('Suite EVA');

    const toggleBtn = element.shadowRoot?.querySelector('.language-toggle');
    expect(toggleBtn?.textContent?.trim()).toBe('English');
  });

  // Mobile responsive tests
  it('should apply mobile styles at small viewport', async () => {
    // Note: This is a visual test, would require viewport resize in real scenario
    const topBar = element.shadowRoot?.querySelector('.top-bar');
    expect(topBar).toBeTruthy();
  });

  // 20+ tests total target
});
